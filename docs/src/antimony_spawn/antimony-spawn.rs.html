1//! This application is an implementation of flatpak-spawn for Antimony.
2//! Some of its arguments are no-ops, and only exist for a compatible interface,
3//! but it's largely feature complete. There are two important distinctions:
4//!
5//! The dichotomy between --sandbox and --host are vastly different. When
6//! running in --sandbox, the requested command is run in a new Antimony
7//! sandbox, constructed from the command line arguments. Because the
8//! environment antimony-spawn runs in (Antimony itself) does not have
9//! an Antimony installation, we have no features or profiles; therefore,
10//! we pass the sandbox's lib folder directly--which has already been
11//! reduced to only needing what the main application and its dependencies
12//! require.
13//!
14//! --host does not run a command on the host. It just runs the command
15//! directly in the sandbox. Allowing execution on the host, even
16//! mediated, is a tremendous vulnerability, and trying to implement it
17//! would require significant effort.
18//!
19//! --watch-bus is a no-op: antimony-spawn will either exit if the child
20//! it runs does, or if it is hit with SIGTERM or SIGINT.
21use anyhow::Result;
22use clap::Parser;
23use nix::unistd::chdir;
24use spawn::Spawner;
25use std::{
26    borrow::Cow,
27    env,
28    os::fd::{FromRawFd, OwnedFd},
29    path::Path,
30};
31
32#[derive(Parser, Debug, Default)]
33#[command(name = "Antimony-Spawn")]
34#[command(version)]
35#[command(about = "An implementation of flatpak-spawn for Antimony")]
36pub struct Cli {
37    /// The command to run.
38    pub command: String,
39
40    /// No-op
41    #[arg(short, long, default_value_t = false)]
42    pub verbose: bool,
43
44    /// FD's to forward to the child
45    #[arg(long)]
46    pub forward_fd: Option<Vec<i32>>,
47
48    /// Clear the environment of the child
49    #[arg(long, default_value_t = false)]
50    pub clear_env: bool,
51
52    /// No-op
53    #[arg(long, default_value_t = false)]
54    pub watch_bus: bool,
55
56    /// Environment variables to pass to the child.
57    #[arg(long)]
58    pub env: Option<Vec<String>>,
59
60    /// No-op. The children never have network access.
61    #[arg(long, default_value_t = false)]
62    pub no_network: bool,
63
64    /// Run in a new sandbox
65    #[arg(long, default_value_t = false)]
66    pub sandbox: bool,
67
68    /// Files to pass ReadWrite
69    #[arg(long)]
70    pub sandbox_expose: Option<Vec<String>>,
71
72    /// Files to pass ReadOnly
73    #[arg(long)]
74    pub sandbox_expose_ro: Option<Vec<String>>,
75
76    /// Run in the current environment.
77    #[arg(long, default_value_t = false)]
78    pub host: bool,
79
80    /// Directory to start the spawned command in.
81    #[arg(long)]
82    pub directory: Option<String>,
83
84    /// Passthrough args.
85    #[arg(trailing_var_arg = true, allow_hyphen_values = true)]
86    pub passthrough: Option<Vec<String>>,
87}
88
89fn main() -> Result<()> {
90    // Set a new AT_HOME in temp.
91    unsafe { env::set_var("AT_HOME", "/tmp/antimony-spawn") };
92
93    // If we're running within Antimony itself
94    if env::var("USER").is_err() {
95        unsafe { env::set_var("USER", "antimony") };
96    }
97
98    rayon::ThreadPoolBuilder::new().build_global()?;
99    let cli = Cli::parse();
100
101    // Change the current directory.
102    if let Some(directory) = cli.directory {
103        chdir(Path::new(&directory))?;
104    }
105
106    let mut handle = if cli.sandbox {
107        // Construct the relevant arguments.
108        let mut args = antimony::cli::run::Args {
109            passthrough: cli.passthrough,
110            ro: cli.sandbox_expose_ro,
111            rw: cli.sandbox_expose,
112            env: cli.env,
113            libraries: Some(vec!["/usr/lib".to_string()]),
114            ..Default::default()
115        };
116
117        let mut info = antimony::setup::setup(Cow::Owned(cli.command), &mut args)?;
118
119        // Forward FDs.
120        if let Some(fds) = cli.forward_fd {
121            for fd in fds {
122                info.handle.fd_i(unsafe { OwnedFd::from_raw_fd(fd) });
123            }
124        }
125
126        // Preserve if want to
127        if !cli.clear_env {
128            info.handle.preserve_env_i(true);
129        }
130
131        // Give the sandbox's library
132        #[rustfmt::skip]
133        info.handle.args_i([
134            "--ro-bind", "/etc", "/etc",
135            "--ro-bind", "/usr/share", "/usr/share",
136        ])?;
137
138        // Post and spawn.
139        info.handle.arg_i(info.profile.app_path(&info.name))?;
140        info.handle.args_i(info.post)?;
141        info.handle.spawn()?
142    } else {
143        // If --host, or no --sandbox, just spawn the command.
144        let mut handle = Spawner::new(cli.command)?;
145        if let Some(passthrough) = cli.passthrough {
146            handle.args_i(passthrough)?;
147        }
148        if let Some(fds) = cli.forward_fd {
149            for fd in fds {
150                handle.fd_i(unsafe { OwnedFd::from_raw_fd(fd) });
151            }
152        }
153        if !cli.clear_env {
154            handle.preserve_env_i(true);
155        }
156        handle.spawn()?
157    };
158
159    handle.wait()?;
160    Ok(())
161}
