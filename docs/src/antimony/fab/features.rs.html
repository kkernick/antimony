1//! The Feature Fabricator composes all defined features defined in the Profile,
2//! recursively analyzes dependencies, strikes conflicts, and merges definitions
3//! into a single, complete Profile.
4//!
5//! Note that due to performance considerations, the feature fabricator is called on
6//! Profile::new(), and is cached separately from the other fabricators as well.

21/// Errors related to feature integration
22#[derive(Debug, Error)]
23pub enum Error {
24    /// Invalid bus name.
25    #[error("Invalid bus name: {0}")]
26    InvalidBus(String),
27
28    /// Feature error.
29    #[error("Feature error: {0}")]
30    Feature(feature::Error),
31}

33/// Replace {} names with the real values in the profile.
34#[inline]
35fn format(mut str: String, map: &Map<&str, String>) -> Result<String, Error> {
36    for (key, val) in map {
37        str = str.replace(key, val);
38    }
39
40    if !str.contains('.') {
41        Err(Error::InvalidBus(str))
42    } else {
43        Ok(str)
44    }
45}

47/// Load a feature from the cache database. This prevents loading the same Feature multiple times.
48#[inline]
49fn load_feature<'a>(name: &str, db: &'a mut Map<String, Feature>) -> Result<&'a Feature, Error> {
50    Ok(db
51        .entry(name.to_string())
52        .or_insert(Feature::new(name).map_err(Error::Feature)?))
53}

55/// Strike a feature from the feature list.
56/// This function operates with the following logic:
57///
58/// 1. If this feature is currently required, it is removed from the requirement list *immediately*
59/// 2. Every feature that this feature depends on is decremented in the requirement list.
60/// 3. If those dependencies were only required by the stricken feature, they are also stricken.
61///
62/// This ensures that when a feature is deemed a conflict with another feature, both it, and any
63/// dependencies that were only required by the now stricken feature, are removed.
64///
65/// Remember, conflicting features supersede everything else. No matter where a conflicting feature
66/// is defined, it will be removed from the set regardless of where the conflict exists, how many
67/// other features rely on, etc.
68fn strike_feature(
69    feature: &str,
70    db: &mut Map<String, Feature>,
71    features: &mut Map<String, u32>,
72) -> Result<(), Error> {
73    // If we required this feature
74    if features.contains_key(feature) {
75        debug!("Striking feature: {feature}");
76
77        // Remove the offending feature immediately.
78        features.swap_remove(feature);
79
80        // Then, grab the features that this feature requires.
81        if let Some(depends) = load_feature(feature, db)?.requires.clone() {
82            for depend in depends {
83                // For each, decrement the dependency count.
84                if let Some(feat) = features.get_mut(&depend) {
85                    *feat -= 1;
86
87                    // If this feature was the only one requiring this dependency, strike it as well.
88                    if *feat < 1 {
89                        strike_feature(&depend, db, features)?;
90                    }
91                }
92            }
93        }
94    }
95    Ok(())
96}

98/// Resolves features. This function recursively resolves each feature, and all the required features
99/// it needs. It also excludes any conflicts, with intelligent dependency sorting.
100fn resolve_feature(
101    feature: &str,
102    db: &mut Map<String, Feature>,
103    features: &mut Map<String, u32>,
104    blacklist: &mut Set<String>,
105    searched: &mut Set<String>,
106) -> Result<(), Error> {
107    // If we haven't search this already.
108    if !searched.contains(feature) && !blacklist.contains(feature) {
109        // Add this feature to our feature list if it doesn't exit.
110        *features.entry(feature.to_string()).or_insert(0) += 1;
111
112        // Add to searched.
113        searched.insert(feature.to_string());
114
115        // Get a copy of the required features, and conflicting features.
116        let (requires, conflicts) = {
117            match load_feature(feature, db) {
118                Ok(feature) => (feature.requires.clone(), feature.conflicts.clone()),
119                Err(_) => (None, None),
120            }
121        };
122
123        // Resolve the requirements.
124        if let Some(requires) = requires {
125            for require in requires {
126                resolve_feature(&require, db, features, blacklist, searched)?;
127            }
128        }
129
130        // Strike out conflicts.
131        if let Some(conflicts) = conflicts {
132            blacklist.extend(conflicts.clone());
133            for conflict in conflicts {
134                if features.contains_key(&conflict) {
135                    strike_feature(&conflict, db, features)?;
136                }
137            }
138        }
139    }
140    Ok(())
141}

143fn resolve_features(
144    features: &Set<String>,
145    conflicts: &Set<String>,
146) -> Result<Vec<Feature>, Error> {
147    let mut feature_list = Map::default();
148    let mut searched = Set::default();
149    let mut blacklist = conflicts.clone();
150    let mut db = Map::default();
151
152    for feat in features {
153        resolve_feature(
154            feat.as_str(),
155            &mut db,
156            &mut feature_list,
157            &mut blacklist,
158            &mut searched,
159        )?;
160    }
161    Ok(feature_list
162        .into_keys()
163        .filter_map(|name| db.swap_remove(&name))
164        .collect())
165}

167fn add_feature(
168    profile: &mut Profile,
169    map: &Map<&str, String>,
170    mut feature: Feature,
171) -> Result<(), Error> {
172    // Conditionals intentionally don't use any schema. It just runs the content through
173    // bash. This lets you do something as simple requiring a certain file via `command`, or
174    // something more complicated. It runs under the
175    if let Some(condition) = feature.conditional.take() {
176        let code = || -> anyhow::Result<i32> {
177            let code = Spawner::abs("/usr/bin/bash")
178                .args(["-c", &condition])?
179                .preserve_env(true)
180                .mode(user::Mode::Real)
181                .output(spawn::StreamMode::Pipe)
182                .error(spawn::StreamMode::Pipe)
183                .spawn()?
184                .wait()?;
185            Ok(code)
186        }();
187
188        match code {
189            Ok(code) => {
190                if code != 0 {
191                    debug!("Condition for feature {} not met", &feature.name);
192                    return Ok(());
193                }
194            }
195            Err(e) => {
196                debug!(
197                    "Failed to check condition for feature {}: {e}",
198                    &feature.name
199                );
200                return Ok(());
201            }
202        }
203    }
204
205    // Caveats don't do anything more than warn. If a feature was truly dangerous enough to warrant an error,
206    // it wouldn't be acceptable to use in Antimony at all. Remember, it can't be any worse than running the
207    // application on the host directly. Even the broad features like `lib` and `bin` restrict access to
208    // other processes, the possibility for root access, etc.
209    if let Some(caveat) = feature.caveat.take() {
210        warn!(
211            "This profile uses a dangerous feature! {}: {}",
212            feature.name, caveat
213        );
214    } else {
215        debug!("Adding feature: {}", feature.name);
216    }
217
218    if let Some(files) = feature.files.take() {
219        let p_files = profile.files.get_or_insert_default();
220
221        let mut direct = files.direct;
222        let p_direct = &mut p_files.direct;
223        for mode in FILE_MODES {
224            if let Some(d_files) = direct.swap_remove(&mode) {
225                p_direct.entry(mode).or_default().extend(d_files);
226            };
227        }
228
229        let mut system = files.platform;
230        let p_sys = &mut p_files.platform;
231        for mode in FILE_MODES {
232            if let Some(sys_files) = system.swap_remove(&mode) {
233                p_sys.entry(mode).or_default().extend(
234                    sys_files
235                        .into_iter()
236                        .map(|s| resolve(Cow::Owned(s)).into_owned()),
237                );
238            }
239        }
240
241        let mut system = files.resources;
242        let p_sys = &mut p_files.resources;
243        for mode in FILE_MODES {
244            if let Some(sys_files) = system.swap_remove(&mode) {
245                p_sys.entry(mode).or_default().extend(
246                    sys_files
247                        .into_iter()
248                        .map(|s| resolve(Cow::Owned(s)).into_owned()),
249                );
250            }
251        }
252
253        let mut user = files.user;
254        let p_user = &mut p_files.user;
255
256        for mode in FILE_MODES {
257            if let Some(user_files) = user.swap_remove(&mode) {
258                p_user.entry(mode).or_default().extend(
259                    user_files
260                        .into_iter()
261                        .map(|s| resolve(Cow::Owned(s)).into_owned()),
262                );
263            }
264        }
265    }
266
267    if let Some(binaries) = feature.binaries.take() {
268        profile.binaries.extend(binaries);
269    }
270    if let Some(libraries) = feature.libraries.take() {
271        profile.libraries.extend(libraries);
272    }
273    if let Some(devices) = feature.devices.take() {
274        profile.devices.extend(devices);
275    }
276    if let Some(namespaces) = feature.namespaces.take() {
277        profile.namespaces.extend(namespaces);
278    }
279    if let Some(args) = feature.sandbox_args.take() {
280        profile.sandbox_args.extend(args);
281    }
282
283    if let Some(ipc) = feature.ipc.take() {
284        let p_ipc = profile.ipc.get_or_insert_default();
285
286        // Features, as the name implies, *add* functionality. Antimony
287        // operates under a secure-default, so things need to be added.
288        // Because single-values can conflict with each other, this
289        // design choice is reflected in setting behavior.
290
291        // Features can enable the global bus flags if they need it,
292        // but they cannot try and restrict the bus if another feature/profile
293        // has turned it on.
294        p_ipc.system_bus = match p_ipc.system_bus {
295            Some(false) | None => ipc.system_bus,
296            Some(true) => Some(true),
297        };
298        p_ipc.user_bus = match p_ipc.user_bus {
299            Some(false) | None => ipc.user_bus,
300            Some(true) => Some(true),
301        };
302
303        // Conversely, if a feature or profile has explicitly set
304        // disable to false for compatibility, you cannot enable it.
305        p_ipc.disable = match p_ipc.disable {
306            Some(true) => {
307                if ipc.disable.is_some() {
308                    ipc.disable
309                } else {
310                    Some(true)
311                }
312            }
313            Some(false) => Some(false),
314            None => ipc.disable,
315        };
316
317        let format_all = |ipc_list: Set<String>| -> Set<String> {
318            ipc_list
319                .into_iter()
320                .filter_map(|f| format(f, map).ok())
321                .collect()
322        };
323
324        if !ipc.portals.is_empty() {
325            p_ipc.portals.extend(ipc.portals);
326        }
327        if !ipc.see.is_empty() {
328            let formatted = format_all(ipc.see);
329            if !formatted.is_empty() {
330                p_ipc.see.extend(formatted);
331            }
332        }
333        if !ipc.talk.is_empty() {
334            let formatted = format_all(ipc.talk);
335            if !formatted.is_empty() {
336                p_ipc.talk.extend(formatted);
337            }
338        }
339        if !ipc.own.is_empty() {
340            let formatted = format_all(ipc.own);
341            if !formatted.is_empty() {
342                p_ipc.own.extend(formatted);
343            }
344        }
345        if !ipc.call.is_empty() {
346            p_ipc.call.extend(ipc.call);
347        }
348    }
349
350    if let Some(env) = feature.environment.take() {
351        for (key, value) in env {
352            profile
353                .environment
354                .insert(key, resolve(Cow::Owned(value)).into_owned());
355        }
356    }
357
358    if let Some(hooks) = feature.hooks.take() {
359        let p_hooks = profile.hooks.get_or_insert_default();
360        if let Some(mut pre) = hooks.pre {
361            p_hooks.pre.get_or_insert_default().append(&mut pre);
362        }
363        if let Some(mut post) = hooks.post {
364            p_hooks.post.get_or_insert_default().append(&mut post)
365        }
366
367        // Yeah, no great answer here. The last feature that sets a parent
368        // hook get it.
369        if p_hooks.parent.is_none() {
370            p_hooks.parent = hooks.parent;
371        }
372    }
373    Ok(())
374}

376pub fn fabricate(profile: &mut Profile, name: &str) -> Result<(), Error> {
377    let mut map = Map::default();
378    map.insert("{name}", name.to_string());
379    map.insert("{desktop}", profile.desktop(name).to_string());
380
381    for feature in resolve_features(&profile.features, &profile.conflicts)? {
382        add_feature(profile, &map, feature)?;
383    }
384    Ok(())
385}
