15/// Localize and bind
16#[inline]
17fn localize(mode: FileMode, file: &str, home: bool, handle: &Spawner, can_try: bool) -> Result<()> {
18    match localize_path(file, home)? {
19        (Some(source), dest) => {
20            handle.args_i([Cow::Borrowed(mode.bind(can_try)), source, Cow::Owned(dest)]);
21        }
22        (None, dest) => {
23            let resolved = if home && !file.starts_with("/home") {
24                Cow::Owned(format!("{}/{file}", HOME.as_str()))
25            } else {
26                Cow::Borrowed(file)
27            };
28            handle.args_i([Cow::Borrowed(mode.bind(true)), resolved, Cow::Owned(dest)]);
29        }
30    }
31    Ok(())
32}

34pub fn fabricate(info: &super::FabInfo) -> Result<()> {
35    let lockdown = info.profile.lockdown.unwrap_or(false);
36
37    if let Some(files) = &info.profile.files {
38        for temp in &files.temp {
39            info.handle.args_i(["--tmpfs", temp]);
40        }
41
42        if !lockdown {
43            let user_files = &files.user;
44            for mode in FILE_MODES {
45                if let Some(files) = user_files.get(&mode) {
46                    files.into_par_iter().try_for_each(|file| {
47                        localize(
48                            mode,
49                            &file.replace('~', HOME.as_str()),
50                            true,
51                            info.handle,
52                            true,
53                        )
54                    })?;
55                }
56            }
57        }
58
59        let system = &files.platform;
60        for mode in FILE_MODES {
61            if let Some(files) = system.get(&mode) {
62                files
63                    .into_par_iter()
64                    .try_for_each(|file| localize(mode, file, false, info.handle, true))?;
65            }
66        }
67
68        let system = &files.resources;
69        for mode in FILE_MODES {
70            if let Some(files) = system.get(&mode) {
71                files
72                    .into_par_iter()
73                    .try_for_each(|file| localize(mode, file, false, info.handle, false))?;
74            }
75        }
76    }
77    Ok(())
78}
