13/// An error for issues around Profiles.
14#[derive(Debug, Error)]
15pub enum Error {
16    /// Misc IO errors.
17    #[error("I/O error: {0}")]
18    Io(#[from] io::Error),
19
20    /// Serialization errors.
21    #[error("Failed to serialize file: {0}")]
22    Serialize(#[from] toml::ser::Error),
23
24    /// Serialization errors.
25    #[error("Failed to deserialize file: {0}")]
26    Deserialize(#[from] toml::de::Error),
27
28    /// Misc Errno errors.
29    #[error("System error: {0}")]
30    Errno(#[from] nix::errno::Errno),
31
32    /// Errors running the profile
33    #[error("Failed to spawn editor: {0}")]
34    Spawn(#[from] spawn::SpawnError),
35
36    /// Errors managing a spawned profile
37    #[error("Failed to communicate with editor: {0}")]
38    Handle(#[from] spawn::HandleError),
39
40    /// Dialog errors
41    #[error("Dialog errors: {0}")]
42    Dialog(#[from] dialoguer::Error),
43}

45/// Edit a file via a temporary, committing the changes back into the file.
46pub fn edit<T: DeserializeOwned + Serialize + PartialEq>(
47    original: &str,
48) -> Result<Option<String>, Error> {
49    let digest = toml::from_str(original)?;
50
51    // Pivot to real mode to edit the temporary.
52    // Editors, like vim, can run arbitrary commands, and we don't want
53    // to extend privilege.
54    let temp = as_real!(Result<_, Error>, {
55        let temp = temp::Builder::new()
56            .owner(Mode::Real)
57            .extension("toml")
58            .create::<temp::File>()?;
59
60        fs::write(temp.full(), original)?;
61        Ok(temp)
62    })??;
63
64    // Loop until the user either:
65    //  1. Provides a valid edit.
66    //  2. Bails
67    let buffer = loop {
68        // Launch the editor.
69        Spawner::new(EDITOR.as_str())?
70            .preserve_env(true)
71            .arg(temp.full().to_string_lossy())?
72            .mode(user::Mode::Real)
73            .spawn()?
74            .wait()?;
75
76        // Read the contents.
77        match fs::read_to_string(temp.full()) {
78            Ok(string) => match toml::from_str::<T>(string.as_ref()) {
79                // If they didn't make any changes, we want to tell edit
80                // so that they don't create a redundant user profile.
81                Ok(profile) => {
82                    if string == original {
83                        println!("No modification made.");
84                        return Ok(None);
85                    } else {
86                        break profile;
87                    }
88                }
89
90                // If there's an error, make the user correct, or bail entirely.
91                Err(e) => {
92                    let retry = Confirm::new()
93                        .with_prompt(format!("Syntax error: {e}\nTry again?"))
94                        .interact()?;
95
96                    if !retry {
97                        return Ok(Some(string));
98                    }
99                }
100            },
101            Err(e) => {
102                error!("Failed to read temporary profile: {e}");
103                return Ok(None);
104            }
105        }
106    };
107
108    Ok(if buffer == digest {
109        None
110    } else {
111        Some(toml::to_string(&buffer)?)
112    })
113}
