18/// An error for issues around Profiles.
19#[derive(Debug, Error)]
20pub enum Error {
21    /// Misc IO errors.
22    #[error("I/O error: {0}")]
23    Io(#[from] io::Error),
24
25    /// Serialization errors.
26    #[error("Failed to serialize file: {0}")]
27    Serialize(#[from] toml::ser::Error),
28
29    /// Misc Errno errors.
30    #[error("System error: {0}")]
31    Errno(#[from] nix::errno::Errno),
32
33    /// Errors running the profile
34    #[error("Failed to spawn editor: {0}")]
35    Spawn(#[from] spawn::SpawnError),
36
37    /// Errors managing a spawned profile
38    #[error("Failed to communicate with editor: {0}")]
39    Handle(#[from] spawn::HandleError),
40
41    /// Dialog errors
42    #[error("Dialog errors: {0}")]
43    Dialog(#[from] dialoguer::Error),
44}

46/// Edit a file via a temporary, committing the changes back into the file.
47pub fn edit<T: DeserializeOwned + Serialize>(path: &Path) -> Result<Option<()>, Error> {
48    // Pivot to real mode to edit the temporary.
49    // Editors, like vim, can run arbitrary commands, and we don't want
50    // to extend privilege.
51    let temp = as_real!(Result<_, Error>, {
52        let temp = temp::Builder::new()
53            .owner(Mode::Real)
54            .extension("toml")
55            .create::<temp::File>()?;
56
57        fs::copy(path, temp.full())?;
58        Ok(temp)
59    })??;
60
61    let original = fs::read_to_string(path)?;
62
63    // Loop until the user either:
64    //  1. Provides a valid edit.
65    //  2. Bails
66    let buffer = loop {
67        // Launch the editor.
68        Spawner::new(EDITOR.as_str())?
69            .preserve_env(true)
70            .arg(temp.full().to_string_lossy())?
71            .mode(user::Mode::Real)
72            .spawn()?
73            .wait()?;
74
75        // Read the contents.
76        match fs::read_to_string(temp.full()) {
77            Ok(string) => match toml::from_str::<T>(string.as_ref()) {
78                // If they didn't make any changes, we want to tell edit
79                // so that they don't create a redundant user profile.
80                Ok(profile) => {
81                    if string == original {
82                        println!("No modification made.");
83                        return Ok(None);
84                    } else {
85                        break profile;
86                    }
87                }
88
89                // If there's an error, make the user correct, or bail entirely.
90                Err(e) => {
91                    let retry = Confirm::new()
92                        .with_prompt(format!("Syntax error: {e}\nTry again?"))
93                        .interact()?;
94
95                    if !retry {
96                        return Ok(Some(()));
97                    }
98                }
99            },
100            Err(e) => {
101                error!("Failed to read temporary profile: {e}");
102                return Ok(None);
103            }
104        }
105    };
106
107    as_effective!(Result<(), Error>, {
108        write!(
109            File::create(path)?,
110            "{}",
111            toml::to_string(&buffer)?
112        )?;
113        Ok(())
114    })??;
115
116    Ok(Some(()))
117}
