16/// An error for issues around Profiles.
17#[derive(Debug)]
18pub enum Error {
19    /// When the profile cannot be Deserialized.
20    Deserialize(toml::de::Error),
21
22    /// When the profile cannot be Serialized.
23    Serialize(toml::ser::Error),
24
25    /// Misc IO errors.
26    Io(&'static str, io::Error),
27
28    /// Misc Errno errors.
29    Errno(&'static str, nix::errno::Errno),
30
31    /// Errors resolving/creating paths.
32    Path(which::Error),
33
34    /// Errors running the profile
35    Spawn(spawn::SpawnError),
36
37    /// Errors managing a spawned profile
38    Handle(spawn::HandleError),
39
40    /// Dialog errors
41    Dialog(dialoguer::Error),
42}
43impl fmt::Display for Error {
44    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
45        match self {
46            Self::Deserialize(e) => write!(f, "Failed to read profile: {e}"),
47            Self::Serialize(e) => write!(f, "Failed to write profile: {e}"),
48            Self::Io(what, e) => write!(f, "Failed to {what}: {e}"),
49            Self::Errno(what, e) => write!(f, "{what} failed: {e}"),
50            Self::Path(e) => write!(f, "Path error: {e}"),
51            Self::Spawn(e) => write!(f, "Failed to run command: {e}"),
52            Self::Handle(e) => write!(f, "Failed to manage subprocess: {e}"),
53            Self::Dialog(e) => write!(f, "Failed to prompt user: {e}"),
54        }
55    }
56}
57impl error::Error for Error {
58    fn source(&self) -> Option<&(dyn error::Error + 'static)> {
59        match self {
60            Self::Deserialize(e) => Some(e),
61            Self::Serialize(e) => Some(e),
62            Self::Io(_, e) => Some(e),
63            Self::Errno(_, e) => Some(e),
64            Self::Path(e) => Some(e),
65            Self::Dialog(e) => Some(e),
66            _ => None,
67        }
68    }
69}

74}

84}
85impl From<toml::de::Error> for Error {
86    fn from(val: toml::de::Error) -> Self {
87        Error::Deserialize(val)
88    }
89}
90impl From<toml::ser::Error> for Error {
91    fn from(val: toml::ser::Error) -> Self {
92        Error::Serialize(val)
93    }
94}
95impl From<dialoguer::Error> for Error {
96    fn from(val: dialoguer::Error) -> Self {
97        Error::Dialog(val)
98    }
99}
100impl From<nix::errno::Errno> for Error {
101    fn from(val: nix::errno::Errno) -> Self {
102        Error::Errno("Generic Error", val)
103    }
104}

106pub fn edit<T: DeserializeOwned + Serialize>(path: &Path) -> Result<Option<()>, Error> {
107    // Pivot to real mode to edit the temporary.
108    // Editors, like vim, can run arbitrary commands, and we don't want
109    // to extend privilege.
110    let temp = as_real!(Result<_, Error>, {
111        let temp = temp::Builder::new()
112            .owner(Mode::Real)
113            .extension("toml")
114            .create::<temp::File>()
115            .map_err(|e| Error::Io("open temporary file", e))?;
116
117        fs::copy(path, temp.full()).map_err(|e| Error::Io("write temporary file", e))?;
118        Ok(temp)
119    })??;
120
121    let original = fs::read_to_string(path).map_err(|e| Error::Io("read original profile", e))?;
122
123    // Loop until the user either:
124    //  1. Provides a valid edit.
125    //  2. Bails
126    let buffer = loop {
127        // Launch the editor.
128        Spawner::new(EDITOR.as_str())?
129            .preserve_env(true)
130            .arg(temp.full().to_string_lossy())?
131            .mode(user::Mode::Real)
132            .spawn()?
133            .wait()?;
134
135        // Read the contents.
136        match fs::read_to_string(temp.full()) {
137            Ok(string) => match toml::from_str::<T>(string.as_ref()) {
138                // If they didn't make any changes, we want to tell edit
139                // so that they don't create a redundant user profile.
140                Ok(profile) => {
141                    if string == original {
142                        println!("No modification made.");
143                        return Ok(None);
144                    } else {
145                        break profile;
146                    }
147                }
148
149                // If there's an error, make the user correct, or bail entirely.
150                Err(e) => {
151                    let retry = Confirm::new()
152                        .with_prompt(format!("Syntax error: {e}\nTry again?"))
153                        .interact()?;
154
155                    if !retry {
156                        return Ok(Some(()));
157                    }
158                }
159            },
160            Err(e) => {
161                error!("Failed to read temporary profile: {e}");
162                return Ok(None);
163            }
164        }
165    };
166
167    as_effective!(Result<(), Error>, {
168        write!(
169            File::create(path).map_err(|e| Error::Io("truncate file", e))?,
170            "{}",
171            toml::to_string(&buffer)?
172        )
173        .map_err(|e| Error::Io("write file", e))?;
174        Ok(())
175    })??;
176
177    Ok(Some(()))
178}
