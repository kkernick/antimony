35/// An error for issues around Profiles.
36#[derive(Debug, Error)]
37pub enum Error {
38    /// When a profile doesn't exist.
39    #[error("No such profile {0}: {1}")]
40    NotFound(String, Cow<'static, str>),
41
42    /// When the profile cannot be Deserialized.
43    #[error("Failed to deserialize profile: {0}")]
44    Deserialize(#[from] toml::de::Error),
45
46    /// When the profile cannot be Deserialized.
47    #[error("Failed to cache profile: {0}")]
48    Cache(#[from] postcard::Error),
49
50    /// When the profile cannot be Serialized.
51    #[error("Failed to serialize profile: {0}")]
52    Serialize(#[from] toml::ser::Error),
53
54    /// Misc IO errors.
55    #[error("I/O Error: {0}")]
56    Io(#[from] io::Error),
57
58    /// Misc Errno errors.
59    #[error("System error: {0}: {1}")]
60    Errno(&'static str, nix::errno::Errno),
61
62    /// Misc Errno errors.
63    #[error("Failed to switch user: {0}")]
64    User(#[from] nix::errno::Errno),
65
66    /// Errors resolving/creating paths.
67    #[error("Path error: {0}")]
68    Path(#[from] which::Error),
69
70    /// Errors for profile arguments specified on the command line.
71    #[error("Command line error: {0}")]
72    CommandLine(&'static str, String, Vec<String>),
73
74    /// Errors incorporating features.
75    #[error("Feature error: {0}")]
76    Feature(#[from] crate::fab::features::Error),
77
78    #[error("Profile Store Error: {0}")]
79    Store(#[from] store::Error),
80}

82/// Append two things together. Used for Profile Merging.
83fn append<T>(s: &mut Option<Vec<T>>, p: Option<Vec<T>>) {
84    if let Some(mut source) = p {
85        if let Some(dest) = s {
86            dest.append(&mut source)
87        } else {
88            *s = Some(source);
89        }
90    }
91}

93/// Print info about the libraries used in a feature/profile.
94pub fn library_info(libraries: &Set<String>, verbose: u8) {
95    println!("\t- Libraries:");
96    for library in libraries {
97        if verbose > 2 && library.contains("*") {
98            match get_wildcards(library, true) {
99                Ok(wilds) => {
100                    for wild in wilds {
101                        println!("\t\t- {}", style(wild).italic());
102                    }
103                }
104                Err(_) => println!("\t\t- {}", style(library).italic()),
105            }
106        } else {
107            println!("\t\t- {}", style(library).italic());
108        }
109    }
110}

112/// The definitions needed to sandbox an application.
113#[derive(Deserialize, Serialize, Default, Debug, PartialEq)]
114#[serde(deny_unknown_fields, default)]
115pub struct Profile {
116    /// The path to the application
117    pub path: Option<String>,
118
119    /// The path to start within inside the sandbox.
120    pub dir: Option<String>,
121
122    /// The ID of the application is a unique identifier that, when not defined,
123    /// defaults to the name of the binary. It should be the name of the associated
124    /// .desktop file installed in /usr/share/applications used to launch the
125    /// program normally.
126    ///
127    /// It's used in two ways:
128    ///     1.  It's used as the Internal Flatpak ID. Gnome sources the icon for an application
129    ///         by loading the desktop file with the same name as the ID (With the .desktop)
130    ///         extension. If it cannot find such a desktop file, it will return a generic icon.
131    ///         Other desktop environments, such as KDE, will use the icon defined in the .desktop
132    ///         file that was launched originally. This means that the internal ID is irrelevant
133    ///         when running under KDE.
134    ///     2.  For integration.
135    pub id: Option<String>,
136
137    /// Features the sandbox uses.
138    #[serde(skip_serializing_if = "Set::is_empty")]
139    pub features: Set<String>,
140
141    /// Features that should be excluded from running under the profile.
142    #[serde(skip_serializing_if = "Set::is_empty")]
143    pub conflicts: Set<String>,
144
145    /// A list of profiles to use as a foundation for missing values.
146    ///
147    /// Missing values inherit those from the inherited profiles,
148    /// conflicting values take the profile's, not the inherited.
149    ///
150    /// This can be used to create multiple variants of a single profiles, such as
151    /// versions of an editor (Zed Preview inherits Zed), a baseline configuration
152    /// (LibreOffice), or different variants (A version of Chromium using the Home,
153    /// and another without any home for a clean variant).
154    ///
155    /// path and id cannot be inherited. However,
156    /// inherit itself is recursive, so if a profile inherits a profile, and that profile
157    /// inherits something else, the top-level profile will inherit both.
158    ///
159    /// When this value is not defined, it will default to ["default"], which will inherit
160    /// the user Default Profile. If you define inherits, if you do not put "default" in the
161    /// list the profile will exclude the default profile (In case you need to exempt a profile
162    /// from the Default Profile). You can define inherits to [] if you just want to exempt
163    /// the Profile from the Default.
164    pub inherits: Option<Set<String>>,
165
166    /// Configuration for the profile's home.
167    pub home: Option<home::Home>,
168
169    /// The SECCOMP policy dictates whether to use SECCOMP to constrain the sandbox.
170    pub seccomp: Option<seccomp::SeccompPolicy>,
171
172    /// IPC communication through D-Bus mediated via xdg-dbus-proxy.
173    pub ipc: Option<ipc::Ipc>,
174
175    /// Files passed to the sandbox. System files are canonicalized at the sandbox root,
176    /// Home files are canonicalized at /home/antimony
177    pub files: Option<files::Files>,
178
179    /// Binaries needed in the sandbox.
180    #[serde(skip_serializing_if = "Set::is_empty")]
181    pub binaries: Set<String>,
182
183    /// Libraries needed in the sandbox. They can be listed as:
184    /// 1. Files (eg /usr/lib/lib.so)
185    /// 2. Directories (eg /usr/lib/mylib) to which all contents will be resolved
186    /// 3. Wildcards (eg lib*), which can match directories and files.
187    #[serde(skip_serializing_if = "Set::is_empty")]
188    pub libraries: Set<String>,
189
190    /// Devices needed in the sandbox, at /dev.
191    #[serde(skip_serializing_if = "Set::is_empty")]
192    pub devices: Set<String>,
193
194    /// Namespaces, such as User and Net.
195    #[serde(skip_serializing_if = "Set::is_empty")]
196    pub namespaces: Set<ns::Namespace>,
197
198    /// Environment Variable Keypairs
199    #[serde(skip_serializing_if = "Map::is_empty")]
200    pub environment: Map<String, String>,
201
202    /// Arguments to pass to the sandboxed application directly.
203    #[serde(skip_serializing_if = "Vec::is_empty")]
204    pub arguments: Vec<String>,
205
206    /// Configurations act as embedded profiles, inheriting the main one.
207    #[serde(skip_serializing_if = "Map::is_empty")]
208    pub configuration: Map<String, Profile>,
209
210    /// Hooks are either embedded shell scripts, or paths to executables that are run in coordination with the profile.
211    pub hooks: Option<hooks::Hooks>,
212
213    /// Whether the program has unique privileges that NO_NEW_PRIVS can restrict.
214    /// Note that this does grant privileges, it merely allows an application with existing privileges to
215    /// keep them when running within the sandbox. However, this property being allowed in the sandbox
216    /// means that an other unprivileged process could gain extra privilege if there's a binary in the
217    /// sandbox with privilege, and this flag is enabled (Though note the sandbox cannot elevate to root,
218    /// regardless of privilege).
219    pub new_privileges: Option<bool>,
220
221    /// Arguments to pass to Bubblewrap directly before the program. This could be actual bubblewrap arguments,
222    /// or a wrapper for the sandbox.
223    #[serde(skip_serializing_if = "Vec::is_empty")]
224    pub sandbox_args: Vec<String>,
225}
226impl Profile {
227    /// Construct a profile from the command line.
228    /// Technically, everything needed for a profile can be specified
229    /// from the command line that is needed to run a profile, so
230    /// this can be used to either supplement a profile, or run applications
231    /// without profiles defined.
232    ///
233    /// You probably shouldn't use this as the default way of running stuff,
234    /// however.
235    pub fn from_args(args: &mut cli::run::Args) -> Result<Self, Error> {
236        let mut profile = Self {
237            path: args.path.take(),
238            dir: args.dir.take(),
239            seccomp: args.seccomp.take(),
240            new_privileges: args.new_privileges.take(),
241            ..Default::default()
242        };
243
244        if let Some(mut arguments) = args.passthrough.take() {
245            profile.arguments.append(&mut arguments);
246        }
247
248        if let Some(mut sandbox_args) = args.sandbox_args.take() {
249            profile.sandbox_args.append(&mut sandbox_args);
250        }
251
252        if let Some(features) = args.features.take() {
253            profile.features.extend(features)
254        }
255
256        if let Some(inherits) = args.inherits.take() {
257            profile.inherits = Some(inherits.into_iter().collect())
258        }
259
260        if let Some(conflicts) = args.conflicts.take() {
261            profile.conflicts.extend(conflicts)
262        }
263
264        if let Some(binaries) = args.binaries.take() {
265            profile.binaries.extend(binaries)
266        }
267
268        if let Some(libraries) = args.libraries.take() {
269            profile.libraries.extend(libraries)
270        }
271
272        if let Some(devices) = args.devices.take() {
273            profile.devices.extend(devices)
274        }
275
276        if let Some(namespaces) = args.namespaces.take() {
277            profile.namespaces.extend(namespaces)
278        }
279
280        profile.files = files::Files::from_args(args);
281        profile.ipc = ipc::Ipc::from_args(args);
282
283        if let Some(env) = args.env.take() {
284            let environment = &mut profile.environment;
285            env.into_iter().for_each(|pair| {
286                if let Some((key, value)) = pair.split_once('=') {
287                    environment.insert(key.to_string(), value.to_string());
288                }
289            });
290        }
291
292        if let Some(lock) = args.home_lock.take() {
293            profile.home.get_or_insert_default().lock = Some(lock);
294        }
295        if let Some(name) = args.home_name.take() {
296            profile.home.get_or_insert_default().name = Some(name);
297        }
298        if let Some(path) = args.home_path.take() {
299            profile.home.get_or_insert_default().path = Some(path);
300        }
301        if let Some(policy) = args.home_policy.take() {
302            profile.home.get_or_insert_default().policy = Some(policy);
303        }
304
305        fab::features::fabricate(&mut profile, "cmdline")?;
306        Ok(profile)
307    }

309    /// Load a new profile from all supported locations.
310    pub fn new(name: &str, config: Option<String>) -> Result<Profile, Error> {
311        debug!("Loading {name}");
312
313        let mut profile = timer!(
314            "::load",
315            store::load::<Self, Error>(name, Object::Profile, true)
316        )?;
317        if name == "default" {
318            return Ok(profile);
319        }
320
321        let hash = timer!("::hash", {
322            let mut hash = profile.hash_str()?;
323            if let Some(config) = &config {
324                hash += config;
325            }
326            hash += &CONFIG_FILE.system_mode().to_string();
327            hash
328        });
329
330        let cache = PROFILE_CACHE.join(hash);
331        if let Ok(str) = fs::read_to_string(&cache) {
332            debug!("Using cached profile");
333            return Ok(timer!("::cache_parse", toml::from_str(&str)?));
334        }
335
336        debug!("No cache available");
337        let to_inherit: Set<String> = match &profile.inherits {
338            Some(i) => i.clone(),
339            None => {
340                if !CONFIG_FILE.system_mode()
341                    && USER_STORE.with_borrow(|s| s.exists("default", Object::Profile))
342                {
343                    Set::from_iter(["default".to_string()])
344                } else {
345                    Set::default()
346                }
347            }
348        };
349
350        for inherit in to_inherit {
351            profile.merge(Profile::new(&inherit, None)?)?;
352        }
353
354        if let Some(config) = config {
355            debug!("Loading configuration");
356            if !profile.configuration.is_empty() {
357                match profile.configuration.swap_remove(&config) {
358                    Some(conf) => {
359                        let hooks = conf.hooks.clone();
360                        profile = profile.base(conf)?;
361                        if let Some(hooks) = hooks
362                            && !hooks.inherit.unwrap_or(true)
363                        {
364                            profile.hooks = Some(hooks);
365                        }
366                    }
367                    None => {
368                        return Err(Error::NotFound(
369                            name.to_string(),
370                            Cow::Owned(format!("Configuration {config} does not exist")),
371                        ));
372                    }
373                }
374            } else {
375                return Err(Error::NotFound(
376                    name.to_string(),
377                    Cow::Borrowed("Profile does not have any configurations"),
378                ));
379            }
380        };
381
382        if let Some(path) = &profile.path {
383            if path.starts_with("~") {
384                profile.path = Some(path.replace("~", HOME.as_str()))
385            } else if path.starts_with("$AT_HOME")
386                && let Some(home) = &profile.home
387            {
388                profile.path = Some(path.replace("$AT_HOME", &home.path(name).to_string_lossy()));
389            }
390        }
391
392        // Try and lookup the path. If it doesn't work, then the corresponding application
393        // isn't installed. This is fine, as long as the user doesn't try and run the profile.
394        if !name.ends_with(".toml") && profile.path.is_none() {
395            profile.path = Some(which::which(&profile.app_path(name))?.to_string());
396        }
397
398        debug!("Fabricating features");
399        fab::features::fabricate(&mut profile, name)?;
400        if let Some(parent) = cache.parent()
401            && !parent.exists()
402        {
403            as_effective!(fs::create_dir_all(parent))??;
404        }
405
406        if let Ok(str) = toml::to_string(&profile) {
407            as_effective!(fs::write(cache, &str))??;
408        }
409        Ok(profile)
410    }

412    /// Use another profile as the base for the caller.
413    /// This function effectively inverts the logic of `merge`:
414    /// The values in the source take precedent, either appending or
415    /// overwriting the caller.
416    ///
417    /// The difference is that values unaffected by `merge` persist in
418    /// the caller.
419    pub fn base(mut self, mut source: Self) -> Result<Self, Error> {
420        source.id = self.id.take();
421        source.inherits = self.inherits.take();
422
423        source.merge(self)?;
424        Ok(source)
425    }

427    /// Merge the contents of one profile into another.
428    /// The merging process follows two rules:
429    ///     1.  If the caller has a value defined for single-value
430    ///         options, the argument's value is ignored.
431    ///     2.  For list values, the argument's values are either
432    ///         taken if the caller has no value defined, or
433    ///         the caller's list is appended with the contents
434    ///         of the argument.
435    /// This function fails if inherit is defined, but its
436    /// contents cannot be inherited.
437    ///
438    /// Cache status is not inherited, neither is path, name,
439    /// or desktop.
440    pub fn merge(&mut self, mut profile: Self) -> Result<(), Error> {
441        if self.path.is_none() {
442            self.path = profile.path;
443        }
444
445        if self.dir.is_none() {
446            self.dir = profile.dir;
447        }
448
449        if self.seccomp.is_none() {
450            self.seccomp = profile.seccomp;
451        }
452
453        if self.new_privileges.is_none() {
454            self.new_privileges = profile.new_privileges;
455        }
456
457        if let Some(home) = profile.home {
458            if let Some(s_home) = &mut self.home {
459                s_home.merge(home)
460            } else {
461                self.home = Some(home)
462            }
463        }
464
465        if let Some(files) = profile.files {
466            if let Some(s_files) = &mut self.files {
467                s_files.merge(files)
468            } else {
469                self.files = Some(files);
470            }
471        }
472
473        if let Some(ipc) = profile.ipc {
474            if let Some(s_ipc) = &mut self.ipc {
475                s_ipc.merge(ipc)
476            } else {
477                self.ipc = Some(ipc);
478            }
479        }
480
481        if let Some(hooks) = profile.hooks {
482            if let Some(s_hooks) = &mut self.hooks {
483                s_hooks.merge(hooks)
484            } else {
485                self.hooks = Some(hooks)
486            }
487        }
488
489        for (name, config) in profile.configuration {
490            self.configuration.insert(name, config);
491        }
492
493        for (key, val) in profile.environment {
494            self.environment.insert(key, val);
495        }
496
497        self.namespaces.extend(profile.namespaces);
498        self.binaries.extend(profile.binaries);
499        self.libraries.extend(profile.libraries);
500        self.devices.extend(profile.devices);
501        self.features.extend(profile.features);
502        self.conflicts.extend(profile.conflicts);
503        self.arguments.append(&mut profile.arguments);
504        self.sandbox_args.append(&mut profile.sandbox_args);
505        Ok(())
506    }

508    pub fn app_path<'a>(&'a self, name: &'a str) -> Cow<'a, str> {
509        match &self.path {
510            Some(path) => Cow::Borrowed(path),
511            None => match which(name) {
512                Ok(path) => Cow::Borrowed(path),
513                Err(_) => Cow::Borrowed(name),
514            },
515        }
516    }

518    /// Get the id name, using as the Flatpak ID.
519    /// It's either the id name, or from name()
520    pub fn desktop<'a, 'b>(&'b self, name: &'a str) -> Cow<'a, str>
521    where
522        'b: 'a,
523    {
524        if let Some(id) = &self.id {
525            Cow::Borrowed(id)
526        } else {
527            let path = self.app_path(name);
528            let bin_name = if let Some(i) = path.rfind('/') {
529                let slice = &path[i + 1..];
530                Cow::Owned(slice.to_string())
531            } else {
532                path
533            };
534
535            if bin_name.contains('.') {
536                bin_name
537            } else {
538                Cow::Borrowed(name)
539            }
540        }
541    }

543    /// Format the id as the Flatpak ID.
544    pub fn id(&self, name: &str) -> String {
545        let id = self.desktop(name);
546        if id.contains('.') {
547            id.to_string()
548        } else {
549            format!("antimony.{id}")
550        }
551    }

553    /// Get the numerical hash of the profile.
554    /// Note that while *deserializing* from postcard throws an error,
555    /// we can serialize it for the purposes of hashing.
556    pub fn num_hash(&self) -> Result<u64, Error> {
557        timer!("::hash", {
558            Ok(RandomState::with_seeds(0, 0, 0, 0).hash_one(&postcard::to_stdvec(&self)?))
559        })
560    }
561
562    /// Get the Profile's hash.
563    pub fn hash_str(&self) -> Result<String, Error> {
564        Ok(format!("{}", self.num_hash()?))
565    }
