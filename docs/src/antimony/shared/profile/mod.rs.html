39/// An error for issues around Profiles.
40#[derive(Debug, Error)]
41pub enum Error {
42    /// When a profile doesn't exist.
43    #[error("No such profile {0}: {1}")]
44    NotFound(String, Cow<'static, str>),
45
46    /// When the profile cannot be Deserialized.
47    #[error("Failed to deserialize profile: {0}")]
48    Deserialize(#[from] toml::de::Error),
49
50    /// When the profile cannot be Deserialized.
51    #[error("Failed to cache profile: {0}")]
52    Cache(#[from] postcard::Error),
53
54    /// When the profile cannot be Serialized.
55    #[error("Failed to serialize profile: {0}")]
56    Serialize(#[from] toml::ser::Error),
57
58    /// Misc IO errors.
59    #[error("I/O Error: {0}: {1}")]
60    Io(&'static str, io::Error),
61
62    /// Misc Errno errors.
63    #[error("System error: {0}: {1}")]
64    Errno(&'static str, nix::errno::Errno),
65
66    /// Misc Errno errors.
67    #[error("Failed to switch user: {0}")]
68    User(#[from] nix::errno::Errno),
69
70    /// Errors resolving/creating paths.
71    #[error("Path error: {0}")]
72    Path(#[from] which::Error),
73
74    /// Errors for profile arguments specified on the command line.
75    #[error("Command line error: {0}")]
76    CommandLine(&'static str, String, Vec<String>),
77
78    /// Errors incorporating features.
79    #[error("Feature error: {0}")]
80    Feature(#[from] crate::fab::features::Error),
81}

83/// Append two things together. Used for Profile Merging.
84fn append<T>(s: &mut Option<Vec<T>>, p: Option<Vec<T>>) {
85    if let Some(mut source) = p {
86        if let Some(dest) = s {
87            dest.append(&mut source)
88        } else {
89            *s = Some(source);
90        }
91    }

94/// Print info about the libraries used in a feature/profile.
95pub fn library_info(libraries: &Set<String>, verbose: u8) {
96    println!("\t- Libraries:");
97    for library in libraries {
98        if verbose > 2 && library.contains("*") {
99            match get_wildcards(library, true) {
100                Ok(wilds) => {
101                    for wild in wilds {
102                        println!("\t\t- {}", style(wild).italic());
103                    }
104                }
105                Err(_) => println!("\t\t- {}", style(library).italic()),
106            }
107        } else {
108            println!("\t\t- {}", style(library).italic());
109        }
110    }
111}

113/// The definitions needed to sandbox an application.
114#[derive(Deserialize, Serialize, Default, Debug)]
115#[serde(deny_unknown_fields, default)]
116pub struct Profile {
117    /// The path to the application
118    pub path: Option<String>,
119
120    /// The ID of the application is a unique identifier that, when not defined,
121    /// defaults to the name of the binary. It should be the name of the associated
122    /// .desktop file installed in /usr/share/applications used to launch the
123    /// program normally.
124    ///
125    /// It's used in two ways:
126    ///     1.  It's used as the Internal Flatpak ID. Gnome sources the icon for an application
127    ///         by loading the desktop file with the same name as the ID (With the .desktop)
128    ///         extension. If it cannot find such a desktop file, it will return a generic icon.
129    ///         Other desktop environments, such as KDE, will use the icon defined in the .desktop
130    ///         file that was launched originally. This means that the internal ID is irrelevant
131    ///         when running under KDE.
132    ///     2.  For integration.
133    pub id: Option<String>,
134
135    /// Features the sandbox uses.
136    #[serde(skip_serializing_if = "Set::is_empty")]
137    pub features: Set<String>,
138
139    /// Features that should be excluded from running under the profile.
140    #[serde(skip_serializing_if = "Set::is_empty")]
141    pub conflicts: Set<String>,
142
143    /// A list of profiles to use as a foundation for missing values.
144    ///
145    /// Missing values inherit those from the inherited profiles,
146    /// conflicting values take the profile's, not the inherited.
147    ///
148    /// This can be used to create multiple variants of a single profiles, such as
149    /// versions of an editor (Zed Preview inherits Zed), a baseline configuration
150    /// (LibreOffice), or different variants (A version of Chromium using the Home,
151    /// and another without any home for a clean variant).
152    ///
153    /// path and id cannot be inherited. However,
154    /// inherit itself is recursive, so if a profile inherits a profile, and that profile
155    /// inherits something else, the top-level profile will inherit both.
156    ///
157    /// When this value is not defined, it will default to ["default"], which will inherit
158    /// the user Default Profile. If you define inherits, if you do not put "default" in the
159    /// list the profile will exclude the default profile (In case you need to exempt a profile
160    /// from the Default Profile). You can define inherits to [] if you just want to exempt
161    /// the Profile from the Default.
162    pub inherits: Option<Set<String>>,
163
164    /// Configuration for the profile's home.
165    pub home: Option<home::Home>,
166
167    /// The SECCOMP policy dictates whether to use SECCOMP to constrain the sandbox.
168    pub seccomp: Option<seccomp::SeccompPolicy>,
169
170    /// IPC communication through D-Bus mediated via xdg-dbus-proxy.
171    pub ipc: Option<ipc::Ipc>,
172
173    /// Files passed to the sandbox. System files are canonicalized at the sandbox root,
174    /// Home files are canonicalized at /home/antimony
175    pub files: Option<files::Files>,
176
177    /// Binaries needed in the sandbox.
178    #[serde(skip_serializing_if = "Set::is_empty")]
179    pub binaries: Set<String>,
180
181    /// Libraries needed in the sandbox. They can be listed as:
182    /// 1. Files (eg /usr/lib/lib.so)
183    /// 2. Directories (eg /usr/lib/mylib) to which all contents will be resolved
184    /// 3. Wildcards (eg lib*), which can match directories and files.
185    #[serde(skip_serializing_if = "Set::is_empty")]
186    pub libraries: Set<String>,
187
188    /// Devices needed in the sandbox, at /dev.
189    #[serde(skip_serializing_if = "Set::is_empty")]
190    pub devices: Set<String>,
191
192    /// Namespaces, such as User and Net.
193    #[serde(skip_serializing_if = "Set::is_empty")]
194    pub namespaces: Set<ns::Namespace>,
195
196    /// Environment Variable Keypairs
197    #[serde(skip_serializing_if = "Map::is_empty")]
198    pub environment: Map<String, String>,
199
200    /// Arguments to pass to the sandboxed application directly.
201    #[serde(skip_serializing_if = "Vec::is_empty")]
202    pub arguments: Vec<String>,
203
204    /// Configurations act as embedded profiles, inheriting the main one.
205    #[serde(skip_serializing_if = "Map::is_empty")]
206    pub configuration: Map<String, Profile>,
207
208    /// Hooks are either embedded shell scripts, or paths to executables that are run in coordination with the profile.
209    pub hooks: Option<hooks::Hooks>,
210
211    /// Whether the program has unique privileges that NO_NEW_PRIVS can restrict.
212    /// Note that this does grant privileges, it merely allows an application with existing privileges to
213    /// keep them when running within the sandbox. However, this property being allowed in the sandbox
214    /// means that an other unprivileged process could gain extra privilege if there's a binary in the
215    /// sandbox with privilege, and this flag is enabled (Though note the sandbox cannot elevate to root,
216    /// regardless of privilege).
217    pub new_privileges: Option<bool>,
218
219    /// Arguments to pass to Bubblewrap directly before the program. This could be actual bubblewrap arguments,
220    /// or a wrapper for the sandbox.
221    #[serde(skip_serializing_if = "Vec::is_empty")]
222    pub sandbox_args: Vec<String>,
223}

237    /// Load a profile from the database. This does not include any feature fabrication.
238    pub fn load(name: &str) -> Result<Self, Error> {
239        if name == "default" {
240            if !CONFIG_FILE.system_mode()
241                && let Ok(str) = fs::read_to_string(Self::user_profile(name))
242            {
243                return Ok(toml::from_str(&str)?);
244            } else {
245                let str = fs::read_to_string(Self::system_profile(name))
246                    .map_err(|e| Error::Io("Reading default", e))?;
247                as_effective!(fs::write(Self::user_profile(name), &str))?
248                    .map_err(|e| Error::Io("Writing default", e))?;
249                return Ok(toml::from_str(&str)?);
250            }
251        }
252
253        // Try and load a file absolutely if the file is given.
254        if name.ends_with(".toml") {
255            let path = PathBuf::from(name);
256            if path.exists() {
257                info!("Using File");
258                return Ok(toml::from_str(
259                    &fs::read_to_string(path).map_err(|e| Error::Io("reading TOML", e))?,
260                )?);
261            }
262        }
263
264        if !CONFIG_FILE.system_mode()
265            && let Ok(str) = fs::read_to_string(Self::user_profile(name))
266        {
267            info!("Using Profile Cache");
268            return Ok(toml::from_str(&str)?);
269        }
270
271        if let Ok(str) = fs::read_to_string(Self::system_profile(name)) {
272            info!("Using System Cache");
273            return Ok(toml::from_str(&str)?);
274        }
275
276        Err(Error::NotFound(
277            name.to_string(),
278            Cow::Borrowed("No such profile"),
279        ))
280    }

282    /// Construct a profile from the command line.
283    /// Technically, everything needed for a profile can be specified
284    /// from the command line that is needed to run a profile, so
285    /// this can be used to either supplement a profile, or run applications
286    /// without profiles defined.
287    ///
288    /// You probably shouldn't use this as the default way of running stuff,
289    /// however.
290    pub fn from_args(args: &mut cli::run::Args) -> Result<Self, Error> {
291        let mut profile = Self {
292            path: args.path.take(),
293            seccomp: args.seccomp.take(),
294            new_privileges: args.new_privileges.take(),
295            ..Default::default()
296        };
297
298        if let Some(mut arguments) = args.passthrough.take() {
299            profile.arguments.append(&mut arguments);
300        }
301
302        if let Some(mut sandbox_args) = args.sandbox_args.take() {
303            profile.sandbox_args.append(&mut sandbox_args);
304        }
305
306        if let Some(features) = args.features.take() {
307            profile.features.extend(features)
308        }
309
310        if let Some(inherits) = args.inherits.take() {
311            profile.inherits = Some(inherits.into_iter().collect())
312        }
313
314        if let Some(conflicts) = args.conflicts.take() {
315            profile.conflicts.extend(conflicts)
316        }
317
318        if let Some(binaries) = args.binaries.take() {
319            profile.binaries.extend(binaries)
320        }
321
322        if let Some(libraries) = args.libraries.take() {
323            profile.libraries.extend(libraries)
324        }
325
326        if let Some(devices) = args.devices.take() {
327            profile.devices.extend(devices)
328        }
329
330        if let Some(namespaces) = args.namespaces.take() {
331            profile.namespaces.extend(namespaces)
332        }
333
334        profile.files = files::Files::from_args(args);
335        profile.ipc = ipc::Ipc::from_args(args);
336
337        if let Some(env) = args.env.take() {
338            let environment = &mut profile.environment;
339            env.into_iter().for_each(|pair| {
340                if let Some((key, value)) = pair.split_once('=') {
341                    environment.insert(key.to_string(), value.to_string());
342                }
343            });
344        }
345
346        if let Some(lock) = args.home_lock.take() {
347            profile.home.get_or_insert_default().lock = Some(lock);
348        }
349        if let Some(name) = args.home_name.take() {
350            profile.home.get_or_insert_default().name = Some(name);
351        }
352        if let Some(path) = args.home_path.take() {
353            profile.home.get_or_insert_default().path = Some(path);
354        }
355        if let Some(policy) = args.home_policy.take() {
356            profile.home.get_or_insert_default().policy = Some(policy);
357        }
358
359        fab::features::fabricate(&mut profile, "cmdline")?;
360        Ok(profile)
361    }

363    /// Load a new profile from all supported locations.
364    pub fn new(name: &str, config: Option<String>) -> Result<Profile, Error> {
365        debug!("Loading {name}");
366
367        let mut profile = timer!("::load", Self::load(name))?;
368        if name == "default" {
369            return Ok(profile);
370        }
371
372        let hash = timer!("::hash", {
373            let mut hash = profile.hash_str()?;
374            if let Some(config) = &config {
375                hash += config;
376            }
377            hash += &CONFIG_FILE.system_mode().to_string();
378            hash
379        });
380
381        let cache = PROFILE_CACHE.join(hash);
382        if let Ok(str) = fs::read_to_string(&cache) {
383            debug!("Using cached profile");
384            return Ok(timer!("::cache_parse", toml::from_str(&str)?));
385        }
386
387        debug!("No cache available");
388        let to_inherit: Set<String> = match &profile.inherits {
389            Some(i) => i.clone(),
390            None => {
391                if !CONFIG_FILE.system_mode() && Self::user_profile("default").exists() {
392                    Set::from_iter(["default".to_string()])
393                } else {
394                    Set::default()
395                }
396            }
397        };
398
399        for inherit in to_inherit {
400            profile.merge(Profile::new(&inherit, None)?)?;
401        }
402
403        if let Some(config) = config {
404            debug!("Loading configuration");
405            if !profile.configuration.is_empty() {
406                match profile.configuration.swap_remove(&config) {
407                    Some(conf) => {
408                        profile = profile.base(conf)?;
409                    }
410                    None => {
411                        return Err(Error::NotFound(
412                            name.to_string(),
413                            Cow::Owned(format!("Configuration {config} does not exist")),
414                        ));
415                    }
416                }
417            } else {
418                return Err(Error::NotFound(
419                    name.to_string(),
420                    Cow::Borrowed("Profile does not have any configurations"),
421                ));
422            }
423        };
424
425        if let Some(path) = &profile.path
426            && path.starts_with("~")
427        {
428            profile.path = Some(path.replace("~", HOME.as_str()))
429        }
430
431        // Try and lookup the path. If it doesn't work, then the corresponding application
432        // isn't installed. This is fine, as long as the user doesn't try and run the profile.
433        if !name.ends_with(".toml") && profile.path.is_none() {
434            profile.path = Some(which::which(&profile.app_path(name))?.to_string());
435        }
436
437        debug!("Fabricating features");
438        fab::features::fabricate(&mut profile, name)?;
439        if let Some(parent) = cache.parent()
440            && !parent.exists()
441        {
442            as_effective!(fs::create_dir_all(parent))?
443                .map_err(|e| Error::Io("create profile cache", e))?;
444        }
445
446        if let Ok(str) = toml::to_string(&profile) {
447            as_effective!(fs::write(cache, &str))?
448                .map_err(|e| Error::Io("write profile cache", e))?;
449        }
450        Ok(profile)
451    }

453    /// Use another profile as the base for the caller.
454    /// This function effectively inverts the logic of `merge`:
455    /// The values in the source take precedent, either appending or
456    /// overwriting the caller.
457    ///
458    /// The difference is that values unaffected by `merge` persist in
459    /// the caller.
460    pub fn base(mut self, mut source: Self) -> Result<Self, Error> {
461        source.id = self.id.take();
462        source.inherits = self.inherits.take();
463
464        source.merge(self)?;
465        Ok(source)
466    }

468    /// Merge the contents of one profile into another.
469    /// The merging process follows two rules:
470    ///     1.  If the caller has a value defined for single-value
471    ///         options, the argument's value is ignored.
472    ///     2.  For list values, the argument's values are either
473    ///         taken if the caller has no value defined, or
474    ///         the caller's list is appended with the contents
475    ///         of the argument.
476    /// This function fails if inherit is defined, but its
477    /// contents cannot be inherited.
478    ///
479    /// Cache status is not inherited, neither is path, name,
480    /// or desktop.
481    pub fn merge(&mut self, mut profile: Self) -> Result<(), Error> {
482        if self.path.is_none() {
483            self.path = profile.path;
484        }
485
486        if self.seccomp.is_none() {
487            self.seccomp = profile.seccomp;
488        }
489
490        if self.new_privileges.is_none() {
491            self.new_privileges = profile.new_privileges;
492        }
493
494        if let Some(home) = profile.home {
495            if let Some(s_home) = &mut self.home {
496                s_home.merge(home)
497            } else {
498                self.home = Some(home)
499            }
500        }
501
502        if let Some(files) = profile.files {
503            if let Some(s_files) = &mut self.files {
504                s_files.merge(files)
505            } else {
506                self.files = Some(files);
507            }
508        }
509
510        if let Some(ipc) = profile.ipc {
511            if let Some(s_ipc) = &mut self.ipc {
512                s_ipc.merge(ipc)
513            } else {
514                self.ipc = Some(ipc);
515            }
516        }
517
518        if let Some(hooks) = profile.hooks {
519            if let Some(s_hooks) = &mut self.hooks {
520                s_hooks.merge(hooks)
521            } else {
522                self.hooks = Some(hooks)
523            }
524        }
525
526        for (name, config) in profile.configuration {
527            self.configuration.insert(name, config);
528        }
529
530        for (key, val) in profile.environment {
531            self.environment.insert(key, val);
532        }
533
534        self.namespaces.extend(profile.namespaces);
535        self.binaries.extend(profile.binaries);
536        self.libraries.extend(profile.libraries);
537        self.devices.extend(profile.devices);
538        self.features.extend(profile.features);
539        self.conflicts.extend(profile.conflicts);
540        self.arguments.append(&mut profile.arguments);
541        self.sandbox_args.append(&mut profile.sandbox_args);
542        Ok(())
543    }

545    pub fn app_path<'a>(&'a self, name: &'a str) -> Cow<'a, str> {
546        match &self.path {
547            Some(path) => Cow::Borrowed(path),
548            None => match which(name) {
549                Ok(path) => Cow::Borrowed(path),
550                Err(_) => Cow::Borrowed(name),
551            },
552        }
553    }

555    /// Get the id name, using as the Flatpak ID.
556    /// It's either the id name, or from name()
557    pub fn desktop<'a, 'b>(&'b self, name: &'a str) -> Cow<'a, str>
558    where
559        'b: 'a,
560    {
561        if let Some(id) = &self.id {
562            Cow::Borrowed(id)
563        } else {
564            let path = self.app_path(name);
565            let bin_name = if let Some(i) = path.rfind('/') {
566                let slice = &path[i + 1..];
567                Cow::Owned(slice.to_string())
568            } else {
569                path
570            };
571
572            if bin_name.contains('.') {
573                bin_name
574            } else {
575                Cow::Borrowed(name)
576            }
577        }
578    }

580    /// Format the id as the Flatpak ID.
581    pub fn id(&self, name: &str) -> String {
582        let id = self.desktop(name);
583        if id.contains('.') {
584            id.to_string()
585        } else {
586            format!("antimony.{id}")
587        }

590    /// Get the numerical hash of the profile.
591    /// Note that while *deserializing* from postcard throws an error,
592    /// we can serialize it for the purposes of hashing.
593    pub fn num_hash(&self) -> Result<u64, Error> {
594        timer!("::hash", {
595            Ok(RandomState::with_seeds(0, 0, 0, 0).hash_one(&postcard::to_stdvec(&self)?))
596        })
597    }
598
599    /// Get the Profile's hash.
600    pub fn hash_str(&self) -> Result<String, Error> {
601        Ok(format!("{}", self.num_hash()?))
602    }
