8/// IPC mediated via xdg-dbus-proxy.
9#[derive(Default, Deserialize, Serialize, PartialEq, Eq, Clone, Debug, Message)]
10#[serde(deny_unknown_fields, default)]
11pub struct Ipc {
12    /// Disable all IPC, regardless of what has been set.
13    pub disable: Option<bool>,
14
15    /// Provide the system bus. Defaults to false
16    pub system_bus: Option<bool>,
17
18    /// Provide the user bus directly. xdg-dbus-proxy is not run. Defaults to false.
19    pub user_bus: Option<bool>,
20
21    /// Freedesktop portals.
22    #[serde(skip_serializing_if = "Set::is_empty")]
23    pub portals: Set<Portal>,
24
25    /// Busses that the sandbox can see, but not interact with.
26    #[serde(skip_serializing_if = "Set::is_empty")]
27    pub sees: Set<String>,
28
29    /// Busses the sandbox can talk over.
30    #[serde(skip_serializing_if = "Set::is_empty")]
31    pub talks: Set<String>,
32
33    /// Busses the sandbox owns.
34    #[serde(skip_serializing_if = "Set::is_empty")]
35    pub owns: Set<String>,
36
37    /// Call semantics.
38    #[serde(skip_serializing_if = "Set::is_empty")]
39    pub calls: Set<String>,
40}
41impl Ipc {
42    /// Merge two IPC sets together.
43    pub fn merge(&mut self, ipc: Self) {
44        if self.disable.is_none() {
45            self.disable = ipc.disable;
46        }
47
48        if self.system_bus.is_none() {
49            self.system_bus = ipc.system_bus;
50        }
51        if self.user_bus.is_none() {
52            self.user_bus = ipc.user_bus;
53        }
54
55        self.portals.extend(ipc.portals);
56        self.sees.extend(ipc.sees);
57        self.talks.extend(ipc.talks);
58        self.owns.extend(ipc.owns);
59        self.calls.extend(ipc.calls);
60    }
61
62    /// Construct an IPC set from the command line.
63    pub fn from_args(args: &mut cli::run::Args) -> Option<Self> {
64        let mut ipc: Option<Self> = None;
65
66        if let Some(portals) = args.portals.take() {
67            ipc.get_or_insert_default().portals = portals.into_iter().collect();
68        }
69        if let Some(see) = args.sees.take() {
70            ipc.get_or_insert_default().sees = see.into_iter().collect();
71        }
72        if let Some(talk) = args.talks.take() {
73            ipc.get_or_insert_default().talks = talk.into_iter().collect();
74        }
75        if let Some(own) = args.owns.take() {
76            ipc.get_or_insert_default().owns = own.into_iter().collect();
77        }
78        if let Some(call) = args.calls.take() {
79            ipc.get_or_insert_default().calls = call.into_iter().collect();
80        }
81
82        if args.user_bus {
83            ipc.get_or_insert_default().user_bus = Some(true);
84        }
85        if args.system_bus {
86            ipc.get_or_insert_default().system_bus = Some(true);
87        }
88        if args.disable_ipc {
89            ipc.get_or_insert_default().disable = Some(true);
90        }
91
92        ipc
93    }
94}

96/// A non-exhaustive list of Portals. Some may not be
97/// implemented for certain Desktop Environments.
98/// Not all applications use portals, even if they
99/// are provided to the sandbox.
100#[derive(Debug, Eq, Hash, PartialEq, Deserialize, Serialize, ValueEnum, Clone, Enumeration)]
101#[serde(deny_unknown_fields)]
102pub enum Portal {
103    Background = 0,
104    Camera = 1,
105    Clipboard = 2,
106    Documents = 3,
107    FileChooser = 4,
108    GlobalShortcuts = 5,
109    Inhibit = 6,
110    Location = 7,
111    Notifications = 8,
112    OpenURI = 9,
113    ProxyResolver = 10,
114    Realtime = 11,
115    ScreenCast = 12,
116    Screenshot = 13,
117    Settings = 14,
118    Secret = 15,
119    NetworkMonitor = 16,
120}
121impl fmt::Display for Portal {
122    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
123        match self {
124            Self::Background => write!(f, "Background"),
125            Self::Camera => write!(f, "Camera"),
126            Self::Clipboard => write!(f, "Clipboard"),
127            Self::Documents => write!(f, "Documents"),
128            Self::FileChooser => write!(f, "FileChooser"),
129            Self::GlobalShortcuts => write!(f, "GlobalShortcuts"),
130            Self::Inhibit => write!(f, "Inhibit"),
131            Self::Location => write!(f, "Location"),
132            Self::Notifications => write!(f, "Notifications"),
133            Self::OpenURI => write!(f, "OpenURI"),
134            Self::ProxyResolver => write!(f, "ProxyResolver"),
135            Self::Realtime => write!(f, "Realtime"),
136            Self::ScreenCast => write!(f, "ScreenCast"),
137            Self::Screenshot => write!(f, "Screenshot"),
138            Self::Settings => write!(f, "Settings"),
139            Self::Secret => write!(f, "Secret"),
140            Self::NetworkMonitor => write!(f, "NetworkMonitor"),
141        }
142    }
