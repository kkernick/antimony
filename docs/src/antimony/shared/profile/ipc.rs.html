8/// IPC mediated via xdg-dbus-proxy.
9#[derive(Default, Deserialize, Serialize, PartialEq, Eq, Clone, Debug)]
10#[serde(deny_unknown_fields, default)]
11pub struct Ipc {
12    /// Disable all IPC, regardless of what has been set.
13    pub disable: Option<bool>,
14
15    /// Provide the system bus. Defaults to false
16    pub system_bus: Option<bool>,
17
18    /// Provide the user bus directly. xdg-dbus-proxy is not run. Defaults to false.
19    pub user_bus: Option<bool>,
20
21    /// Freedesktop portals.
22    #[serde(skip_serializing_if = "Set::is_empty")]
23    pub portals: Set<Portal>,
24
25    /// Busses that the sandbox can see, but not interact with.
26    #[serde(skip_serializing_if = "Set::is_empty")]
27    pub see: Set<String>,
28
29    /// Busses the sandbox can talk over.
30    #[serde(skip_serializing_if = "Set::is_empty")]
31    pub talk: Set<String>,
32
33    /// Busses the sandbox owns.
34    #[serde(skip_serializing_if = "Set::is_empty")]
35    pub own: Set<String>,
36
37    /// Call semantics.
38    #[serde(skip_serializing_if = "Set::is_empty")]
39    pub call: Set<String>,
40}
41impl Ipc {
42    /// Merge two IPC sets together.
43    pub fn merge(&mut self, ipc: Self) {
44        if self.disable.is_none() {
45            self.disable = ipc.disable;
46        }
47
48        if self.system_bus.is_none() {
49            self.system_bus = ipc.system_bus;
50        }
51        if self.user_bus.is_none() {
52            self.user_bus = ipc.user_bus;
53        }
54
55        self.portals.extend(ipc.portals);
56        self.see.extend(ipc.see);
57        self.talk.extend(ipc.talk);
58        self.own.extend(ipc.own);
59        self.call.extend(ipc.call);
60    }
61
62    /// Construct an IPC set from the command line.
63    pub fn from_args(args: &mut cli::run::Args) -> Option<Self> {
64        let mut ipc: Option<Self> = None;
65
66        if let Some(portals) = args.portals.take() {
67            ipc.get_or_insert_default().portals = portals.into_iter().collect();
68        };
69        if let Some(see) = args.see.take() {
70            ipc.get_or_insert_default().see = see.into_iter().collect();
71        };
72        if let Some(talk) = args.talk.take() {
73            ipc.get_or_insert_default().talk = talk.into_iter().collect();
74        };
75        if let Some(own) = args.own.take() {
76            ipc.get_or_insert_default().own = own.into_iter().collect();
77        };
78        if let Some(call) = args.call.take() {
79            ipc.get_or_insert_default().call = call.into_iter().collect();
80        };
81
82        if args.user_bus {
83            ipc.get_or_insert_default().user_bus = Some(true);
84        }
85        if args.system_bus {
86            ipc.get_or_insert_default().system_bus = Some(true);
87        }
88        if args.disable_ipc {
89            ipc.get_or_insert_default().disable = Some(true);
90        }
91
92        ipc
93    }
94
95    /// Get info about the IPC set.
96    pub fn info(&self) {
97        println!("\t- IPC mediated via xdg-dbus-proxy");
98        if !self.portals.is_empty() {
99            println!("\t\t- Portals: {}", format_iter(self.portals.iter()));
100        }
101        if !self.talk.is_empty() {
102            println!("\t\t- Talk: {}", format_iter(self.talk.iter()));
103        }
104        if !self.see.is_empty() {
105            println!("\t\t- Visible: {}", format_iter(self.see.iter()));
106        }
107        if !self.own.is_empty() {
108            println!("\t\t- Owns: {}", format_iter(self.own.iter()));
109        }
110        if !self.call.is_empty() {
111            println!("\t\t- Calls via: {}", format_iter(self.call.iter()));
112        }
113    }
114}

116/// A non-exhaustive list of Portals. Some may not be
117/// implemented for certain Desktop Environments.
118/// Not all applications use portals, even if they
119/// are provided to the sandbox.
120#[derive(Debug, Eq, Hash, PartialEq, Deserialize, Serialize, ValueEnum, Clone)]
121#[serde(deny_unknown_fields)]
122pub enum Portal {
123    Background,
124    Camera,
125    Clipboard,
126    Documents,
127    FileChooser,
128    GlobalShortcuts,
129    Inhibit,
130    Location,
131    Notifications,
132    OpenURI,
133    ProxyResolver,
134    Realtime,
135    ScreenCast,
136    Screenshot,
137    Settings,
138    Secret,
139    NetworkMonitor,
140}
141impl std::fmt::Display for Portal {
142    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
143        match self {
144            Portal::Background => write!(f, "Background"),
145            Portal::Camera => write!(f, "Camera"),
146            Portal::Clipboard => write!(f, "Clipboard"),
147            Portal::Documents => write!(f, "Documents"),
148            Portal::FileChooser => write!(f, "FileChooser"),
149            Portal::GlobalShortcuts => write!(f, "GlobalShortcuts"),
150            Portal::Inhibit => write!(f, "Inhibit"),
151            Portal::Location => write!(f, "Location"),
152            Portal::Notifications => write!(f, "Notifications"),
153            Portal::OpenURI => write!(f, "OpenURI"),
154            Portal::ProxyResolver => write!(f, "ProxyResolver"),
155            Portal::Realtime => write!(f, "Realtime"),
156            Portal::ScreenCast => write!(f, "ScreenCast"),
157            Portal::Screenshot => write!(f, "Screenshot"),
158            Portal::Settings => write!(f, "Settings"),
159            Portal::Secret => write!(f, "Secret"),
160            Portal::NetworkMonitor => write!(f, "NetworkMonitor"),
161        }
162    }
