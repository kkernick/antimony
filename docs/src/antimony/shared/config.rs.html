1//! The configuration file is a global configuration for Antimony.
2
3use crate::shared::{
4    Set, edit,
5    env::{AT_HOME, USER_NAME},
6    store,
7};
8use serde::{Deserialize, Serialize};
9use std::{
10    fs::{self, read_to_string},
11    path::PathBuf,
12    sync::LazyLock,
13};
14use user::as_effective;
15
16pub static CONFIG_PATH: LazyLock<PathBuf> = LazyLock::new(|| AT_HOME.join("config.toml"));
17pub static CONFIG_FILE: LazyLock<ConfigFile> = LazyLock::new(ConfigFile::default);
18
19#[derive(Deserialize, Serialize, PartialEq, Clone)]
20pub struct ConfigFile {
21    pub force_temp: Option<bool>,
22    pub system_mode: Option<bool>,
23    pub auto_refresh: Option<bool>,
24    pub privileged_users: Option<Set<String>>,
25
26    pub config_store: Option<store::Store>,
27    pub cache_store: Option<store::Store>,
28}
29impl ConfigFile {
30    pub fn auto_refresh(&self) -> bool {
31        self.auto_refresh.unwrap_or(false)
32    }
33
34    pub fn force_temp(&self) -> bool {
35        self.force_temp.unwrap_or(false)
36    }
37
38    pub fn system_mode(&self) -> bool {
39        self.system_mode.unwrap_or(false)
40    }
41
42    pub fn is_privileged(&self) -> bool {
43        if let Some(users) = &self.privileged_users {
44            users.contains(USER_NAME.as_str())
45        } else {
46            false
47        }
48    }
49
50    pub fn cache_store(&self) -> store::Store {
51        self.cache_store.unwrap_or(store::Store::File)
52    }
53
54    pub fn config_store(&self) -> store::Store {
55        self.config_store.unwrap_or(store::Store::File)
56    }
57
58    pub fn edit(config: &str) -> Result<Option<String>, edit::Error> {
59        edit::edit::<Self>(config)
60    }
61
62    pub fn update(&self) -> anyhow::Result<()> {
63        as_effective!(anyhow::Result<()>, {
64            fs::write(AT_HOME.join("config.toml"), toml::to_string(self)?)?;
65            Ok(())
66        })??;
67        Ok(())
68    }
69}
70impl Default for ConfigFile {
71    fn default() -> Self {
72        let config_path = AT_HOME.join("config.toml");
73        let mut config = if config_path.exists()
74            && let Ok(content) = read_to_string(config_path)
75            && let Ok(parsed) = toml::from_str(&content)
76        {
77            parsed
78        } else {
79            Self {
80                force_temp: None,
81                system_mode: None,
82                auto_refresh: None,
83                privileged_users: None,
84                cache_store: None,
85                config_store: None,
86            }
87        };
88
89        if let Ok(env) = std::env::var("AT_FORCE_TEMP") {
90            config.force_temp = Some(env != "0")
91        }
92        if let Ok(env) = std::env::var("AT_SYSTEM_MODE") {
93            config.system_mode = Some(env != "0")
94        }
95        if let Ok(env) = std::env::var("AT_AUTO_REFRESH") {
96            config.auto_refresh = Some(env != "0")
97        }
98        if let Ok(env) = std::env::var("AT_CACHE_DB")
99            && env != "0"
100        {
101            config.cache_store = Some(store::Store::Database);
102        }
103        if let Ok(env) = std::env::var("AT_CONFIG_DB")
104            && env != "0"
105        {
106            config.config_store = Some(store::Store::Database);
107        }
108        config
109    }
110}
