1use crate::shared::{edit, env::AT_HOME};
2use nix::libc::getpwuid;
3use serde::{Deserialize, Serialize};
4use std::{collections::BTreeSet, ffi::CString, fs::read_to_string, path::Path, sync::LazyLock};
5use user::USER;
6
7pub static CONFIG_FILE: LazyLock<ConfigFile> = LazyLock::new(ConfigFile::default);
8
9#[derive(Deserialize, Serialize)]
10pub struct ConfigFile {
11    force_temp: Option<bool>,
12    system_mode: Option<bool>,
13    privileged_users: Option<BTreeSet<String>>,
14}
15impl ConfigFile {
16    pub fn force_temp(&self) -> bool {
17        self.force_temp.unwrap_or(false)
18    }
19
20    pub fn system_mode(&self) -> bool {
21        self.system_mode.unwrap_or(false)
22    }
23
24    pub fn is_privileged(&self) -> bool {
25        if let Some(users) = &self.privileged_users {
26            unsafe {
27                let passwd = getpwuid(USER.real.as_raw());
28                if passwd.is_null() || (*passwd).pw_name.is_null() {
29                    false
30                } else {
31                    let name = CString::from_raw((*passwd).pw_name);
32                    let str = name.to_string_lossy();
33                    users.contains(str.as_ref())
34                }
35            }
36        } else {
37            false
38        }
39    }
40
41    pub fn edit(path: &Path) -> Result<Option<()>, edit::Error> {
42        edit::edit::<Self>(path)
43    }
44}
45impl Default for ConfigFile {
46    fn default() -> Self {
47        let config_path = AT_HOME.join("config.toml");
48        let mut config = if config_path.exists()
49            && let Ok(content) = read_to_string(config_path)
50            && let Ok(parsed) = toml::from_str(&content)
51        {
52            parsed
53        } else {
54            Self {
55                ..Default::default()
56            }
57        };
58
59        if let Ok(env) = std::env::var("AT_FORCE_TEMP") {
60            config.force_temp = Some(env != "0")
61        }
62        if let Ok(env) = std::env::var("AT_SYSTEM_MODE") {
63            config.system_mode = Some(env != "0")
64        }
65
66        config
67    }
68}
