1//! Refresh installed profiles.
2
3use crate::{
4    cli::{self, run, run_vec},
5    shared::env::{CACHE_DIR, HOME_PATH},
6};
7use anyhow::Result;
8use indicatif::{ProgressBar, ProgressStyle};
9use log::debug;
10use std::{fs, time::Duration};
11use user::as_real;
12
13#[derive(clap::Args, Default)]
14pub struct Args {
15    /// Run a profile, but refresh its contents.
16    /// If not defined, all profiles are refreshed, but nothing is run.
17    pub profile: Option<String>,
18
19    /// Just delete the cache, don't repopulate.
20    #[arg(short, long, default_value_t = false)]
21    pub dry: bool,
22
23    /// Delete the entire Cache directory. Will break any instance currently running!
24    #[arg(long, default_value_t = false)]
25    pub hard: bool,
26
27    /// Run arguments
28    #[arg(trailing_var_arg = true, allow_hyphen_values = true)]
29    pub passthrough: Option<Vec<String>>,
30}
31impl cli::Run for Args {
32    fn run(self) -> Result<()> {
33        user::set(user::Mode::Effective)?;
34
35        if self.hard {
36            for cache in fs::read_dir(CACHE_DIR.as_path())? {
37                let cache = cache?.path();
38                if cache.is_dir() {
39                    fs::remove_dir_all(cache)?;
40                } else {
41                    fs::remove_file(cache)?;
42                }
43            }
44        } else {
45            let _ = fs::remove_dir_all(CACHE_DIR.join(".lib"));
46            let _ = fs::remove_dir_all(CACHE_DIR.join(".proxy"));
47            let _ = fs::remove_dir_all(CACHE_DIR.join(".direct"));
48            let _ = fs::remove_dir_all(CACHE_DIR.join(".seccomp"));
49            let _ = fs::remove_dir_all(CACHE_DIR.join("binaries"));
50            let _ = fs::remove_dir_all(CACHE_DIR.join("libraries"));
51            let _ = fs::remove_dir_all(CACHE_DIR.join("directories"));
52            let _ = fs::remove_dir_all(CACHE_DIR.join("profiles"));
53            let _ = fs::remove_dir_all(CACHE_DIR.join("wildcards"));
54
55            for cache in fs::read_dir(CACHE_DIR.as_path())? {
56                let cache = cache?;
57                let instances = cache.path().join("instances");
58                if instances.exists() {
59                    for instance in fs::read_dir(&instances)? {
60                        let instance = instance?.path();
61                        match fs::read_link(&instance) {
62                            Ok(e) if !e.exists() => fs::remove_file(&instance)?,
63                            Err(_) => fs::remove_file(&instance)?,
64                            _ => {}
65                        }
66                    }
67                    if fs::read_dir(instances)?.count() == 0 {
68                        fs::remove_dir_all(cache.path())?;
69                    }
70                }
71            }
72        }
73
74        // If a single profile exist, refresh it and it alone.
75        if let Some(profile) = self.profile {
76            let mut args = if let Some(passthrough) = self.passthrough {
77                run_vec(&profile, passthrough)
78            } else {
79                run::Args::default()
80            };
81
82            args.refresh = true;
83            args.dry = self.dry;
84            args.profile = profile.clone();
85            args.run()?;
86
87        // If not dry, repopulate the cache.
88        } else if !self.dry {
89            let profiles = installed_profiles()?;
90
91            // DO NOT TRY AND RUN THIS IN PARALLEL. ANTIMONY WILL
92            // CAUSE A KERNEL PANIC IF YOU RUN IT IN PARALLEL!
93            let pb = ProgressBar::new(profiles.len() as u64);
94            pb.set_style(
95                ProgressStyle::default_spinner()
96                    .template(" {spinner} {msg} [{wide_bar}] {eta_precise} ")?,
97            );
98            pb.enable_steady_tick(Duration::from_millis(100));
99            pb.wrap_iter(profiles.into_iter())
100                .try_for_each(|name| -> Result<()> {
101                    pb.set_message(format!("Refreshing {name}"));
102
103                    let args = run::Args {
104                        profile: name.clone(),
105                        dry: true,
106                        refresh: true,
107                        ..Default::default()
108                    };
109
110                    user::set(user::Mode::Effective)?;
111                    args.run()?;
112                    Ok(())
113                })?;
114        }
115        Ok(())
116    }
117}

119pub fn installed_profiles() -> Result<Vec<String>> {
120    let profiles: Vec<String> = as_real!(Result<Vec<String>>, {
121        let bin = HOME_PATH.join(".local").join("bin");
122        debug!("Refreshing local binaries");
123
124        Ok(fs::read_dir(bin)?
125            .filter_map(|file| {
126                if let Ok(file) = file {
127                    let file = file.path();
128                    if let Ok(dest) = fs::read_link(&file)
129                        && dest.ends_with("antimony")
130                        && let Some(name) = file.file_name()
131                    {
132                        let name = name.to_string_lossy();
133                        return Some(name.into_owned());
134                    }
135                }
136                None
137            })
138            .collect())
139    })??;
140    Ok(profiles)
