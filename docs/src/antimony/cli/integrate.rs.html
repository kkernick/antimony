22#[derive(clap::Args, Debug, Default)]
23pub struct Args {
24    /// The name of the profile
25    pub profile: String,
26
27    /// Undo integration for the profile.
28    #[arg(short, long, default_value_t = false)]
29    pub remove: bool,
30
31    /// Some desktop environments, particularly Gnome, source
32    /// their icons via the Flatpak ID (The Profile ID) in this case.
33    /// This value must be in reverse DNS format, and Antimony automatically
34    /// prepends "antimony." on those that don't. This presents an
35    /// incongruity between ID and desktop that requires a shadow that
36    /// hides the original. If an integrated profile lacks an icon, you
37    /// may need to use this option.
38    #[arg(short, long, default_value_t = false)]
39    pub shadow: bool,
40
41    /// How to integrate configurations
42    #[arg(short, long)]
43    pub config_mode: Option<ConfigMode>,
44
45    /// Create a desktop file if one does not exist
46    #[arg(long, default_value_t = false)]
47    pub create_desktop: bool,
48}

50#[derive(Default, ValueEnum, Copy, Clone, Debug, PartialEq, Eq)]
51pub enum ConfigMode {
52    ///Integrate each configuration as a separate desktop action
53    /// within the main Desktop File.
54    #[default]
55    Action,
56
57    /// Separate each configuration into its own Desktop File. This
58    /// can be useful, say, for setting configurations as default
59    /// application handlers.
60    File,
61}
62
63impl cli::Run for Args {
64    fn run(self) -> Result<()> {
65        user::drop(user::Mode::Real)?;
66        if self.remove {
67            remove(self)
68        } else {
69            integrate(self)
70        }
71    }
72}

74/// Undo integration.
75pub fn remove(cmd: Args) -> Result<()> {
76    let profile = Profile::new(&cmd.profile, None)?;
77
78    let name = &cmd.profile;
79
80    if fs::remove_file(HOME_PATH.join(".local").join("bin").join(name)).is_err() {
81        warn!("Binary does not exist");
82    }
83
84    let application = DATA_HOME
85        .join("applications")
86        .join(format!("{}.desktop", profile.desktop(name)));
87    if fs::remove_file(&application).is_err() {
88        warn!(
89            "Original .desktop file ({}) does not exist. You may need to provide if it differs from the profile name",
90            application.display()
91        );
92    }
93
94    let xdg = CONFIG_HOME
95        .join("autostart")
96        .join(format!("{}.desktop", profile.desktop(name)));
97    let _ = fs::remove_file(&xdg);
98
99    let name = if profile.id(name) != profile.desktop(name) && cmd.shadow {
100        Cow::Owned(profile.id(name))
101    } else {
102        profile.desktop(name)
103    };
104
105    if let Some(configs) = &profile.configuration {
106        for config in configs.keys() {
107            let config_desktop = DATA_HOME
108                .join("applications")
109                .join(format!("{name}-{config}.desktop"));
110            if config_desktop.exists() {
111                fs::remove_file(config_desktop)?;
112            }
113        }
114    }
115
116    let copy = DATA_HOME
117        .join("applications")
118        .join(format!("{name}.desktop"));
119    if fs::remove_file(&copy).is_err() {
120        warn!("Profile .desktop file ({}) does not exist", copy.display());
121    }
122
123    Ok(())
124}

126fn fix_exec(name: &str, local: &str, line: &mut String, config: Option<&str>, cmd: &Args) {
127    let args = match line.find(' ') {
128        Some(index) => &line[index + 1..],
129        None => " ",
130    };
131
132    match config {
133        Some(config) => {
134            *line = format!(
135                "{name}=antimony run {} --config {config} {args}",
136                cmd.profile
137            )
138            .trim()
139            .to_string()
140        }
141        None => *line = format!("{name}={local} {args}").trim().to_string(),
142    }
143}

145fn manage_configurations(
146    contents: &mut Vec<String>,
147    cmd: &Args,
148    profile: &Profile,
149    name: &str,
150    local: &str,
151) -> Result<()> {
152    if let Some(configs) = &profile.configuration {
153        match cmd.config_mode.unwrap_or_default() {
154            // Integrate each into one desktop file with actions.
155            ConfigMode::Action => {
156                contents.push("\n".into());
157                for config in configs.keys() {
158                    contents.push(format!(
159                        "[Desktop Action {config}]\n\
160                        Name=Antimony {} Configuration \n\
161                        Exec=antimony run {} --config {config} %U \n",
162                        config.to_title_case(),
163                        cmd.profile
164                    ));
165                }
166            }
167
168            // Provide each configuration as a desktop file.
169            ConfigMode::File => {
170                for config in configs.keys() {
171                    let config_desktop = DATA_HOME
172                        .join("applications")
173                        .join(format!("{name}-{config}.desktop"));
174
175                    let mut contents = contents.clone();
176                    for line in &mut contents {
177                        // Point to the symlink
178                        if line.starts_with("Exec=") {
179                            fix_exec("Exec", local, line, Some(config.as_str()), cmd);
180                        }
181                        if line.starts_with("TryExec=") {
182                            fix_exec("TryExec", local, line, Some(config.as_str()), cmd);
183                        }
184                        if line.starts_with("Name=") {
185                            line.push_str(&format!(" ({})", config.to_title_case()));
186                        }
187                    }
188
189                    write!(File::create(config_desktop)?, "{}", contents.join("\n"))?;
190                }
191            }
192        }
193    }
194    Ok(())
195}

197/// Format a desktop file to point to Antimony
198fn format_desktop(
199    cmd: &Args,
200    profile: &Profile,
201    name: &str,
202    desktop_file: &Path,
203    local: &str,
204    out_path: &Path,
205) -> Result<()> {
206    let name = if profile.id(name) != profile.desktop(name) && cmd.shadow {
207        // Create a shadow copy to turn on NoDisplay
208        let local_desktop: Vec<String> = fs::read_to_string(desktop_file)
209            .with_context(|| "Failed to read desktop file")?
210            .split("\n")
211            .map(|e| {
212                if e.contains("Name=") {
213                    return e.to_string() + " (Native)\nNoDisplay=true";
214                }
215                e.to_string()
216            })
217            .collect();
218
219        // Write it out.
220        let shadow = out_path.join(format!("{}.desktop", profile.desktop(name)));
221
222        if let Some(parent) = shadow.parent()
223            && !parent.exists()
224        {
225            fs::create_dir_all(parent)?;
226        }
227
228        debug!("Writing shadow");
229        write!(File::create(shadow)?, "{}", local_desktop.join("\n"))?;
230        Cow::Owned(profile.id(name))
231    } else {
232        profile.desktop(name)
233    };
234
235    let desktop =
236        fs::read_to_string(desktop_file).with_context(|| "Failed to read desktop file")?;
237
238    // Make the new desktop file.
239    debug!("Creating desktop file");
240
241    let desktop_actions = desktop.contains("Actions=");
242    let append_actions = |line: &mut String| {
243        if let Some(configs) = &profile.configuration {
244            for name in configs.keys() {
245                line.push_str(&format!("{name};"));
246            }
247        }
248        line.push_str("native;");
249    };
250
251    let mut contents: Vec<String> = desktop.lines().map(|e| e.to_string()).collect();
252    for line in &mut contents {
253        // Point to the symlink
254        if line.starts_with("Exec=") {
255            fix_exec("Exec", local, line, None, cmd);
256            if !desktop_actions {
257                line.push_str("\nActions=");
258                append_actions(line);
259            }
260        }
261        if line.starts_with("TryExec=") {
262            line.clear();
263        }
264
265        // GTK applications subscribe to a single instance paradigm, where
266        // a single instance will spawn new windows, rather than entirely new
267        // instances of the app. However, this just causes D-Bus to launch the
268        // application unconfined.
269        if line.starts_with("DBusActivatable=") {
270            *line = "DBusActivatable=false".to_string();
271        }
272
273        if cmd.config_mode.unwrap_or_default() == ConfigMode::Action && line.starts_with("Actions=")
274        {
275            append_actions(line);
276        }
277    }
278
279    contents.push(format!(
280        "\n[Desktop Action native]\n\
281            Name=Run {} Natively\n\
282            Exec={}",
283        cmd.profile.to_title_case(),
284        profile.app_path(&cmd.profile)
285    ));
286
287    // Add configurations.
288    manage_configurations(&mut contents, cmd, profile, &name, local)?;
289
290    let antimony_desktop = out_path.join(format!("{name}.desktop"));
291    if let Some(parent) = antimony_desktop.parent() {
292        fs::create_dir_all(parent)?;
293    }
294
295    write!(
296        File::create(antimony_desktop).with_context(|| "Failed to create new desktop file")?,
297        "{}",
298        contents.join("\n")
299    )
300    .with_context(|| "Failed to write new desktop file")?;
301    Ok(())
302}

304/// Integrate a profile so it can be launched in place of the original in Desktop Environments.
305pub fn integrate(cmd: Args) -> Result<()> {
306    let profile = Profile::new(&cmd.profile, None)?;
307
308    // Collect environment.
309    let antimony = which::which("antimony")?;
310    let name = &cmd.profile;
311
312    // If ~/.local/bin is in PATH, the symlink takes precedence over
313    // and thus applications will run in the sandbox unless the absolute
314    // path in /usr/bin is given.
315    debug!("Creating symlink in ~/.local/bin");
316    let local = HOME_PATH.join(".local").join("bin");
317    if !local.exists() {
318        println!("Creating a local bin folder at ~/.local/bin. You may need to update your PATH if you want
319            to launch sandboxed applications from the command line without the explicit path.");
320        fs::create_dir_all(&local)?;
321    }
322
323    let local = local.join(name);
324    if !local.exists() {
325        symlink(antimony, &local).with_context(|| "Failed to create local symlink")?;
326    }
327    let local = local.to_string_lossy();
328
329    let desktop_file = Path::new("/usr")
330        .join("share")
331        .join("applications")
332        .join(format!("{}.desktop", profile.desktop(name)));
333    if desktop_file.exists() {
334        info!("Overriding Desktop File");
335        format_desktop(
336            &cmd,
337            &profile,
338            name,
339            &desktop_file,
340            &local,
341            &DATA_HOME.join("applications"),
342        )?;
343    } else if cmd.create_desktop {
344        let mut contents: Vec<String> = vec![
345            "[Desktop Entry]",
346            &format!("Name={name}"),
347            &format!("Exec={local}"),
348            "Type=Application",
349        ]
350        .into_iter()
351        .map(|e| e.to_string())
352        .collect();
353
354        let mut actions = "Actions=native;".to_string();
355        if let Some(config) = &profile.configuration
356            && let Some(ConfigMode::Action) = cmd.config_mode
357        {
358            for config in config.keys() {
359                actions.push_str(&format!("{config};"))
360            }
361        }
362        contents.push(actions);
363
364        contents.extend([
365            "[Desktop Action native]".to_string(),
366            format!("Name=Run {} Natively", name.to_title_case()),
367            format!("Exec={}", profile.app_path(&cmd.profile)),
368        ]);
369
370        let out = DATA_HOME
371            .join("applications")
372            .join(name)
373            .with_extension("desktop");
374        if let Some(parent) = out.parent()
375            && !parent.exists()
376        {
377            fs::create_dir_all(parent)?;
378        }
379        manage_configurations(&mut contents, &cmd, &profile, name, &local)?;
380        fs::write(out, contents.join("\n"))?;
381    }
382
383    let service_file = Path::new("/etc")
384        .join("xdg")
385        .join("autostart")
386        .join(format!("{}.desktop", profile.desktop(name)));
387    if service_file.exists() {
388        info!("Overriding XDG Service");
389        format_desktop(
390            &cmd,
391            &profile,
392            name,
393            &service_file,
394            &local,
395            &CONFIG_HOME.join("autostart"),
396        )?;
397    }
398
399    Ok(())
