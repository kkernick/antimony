20/// Create run arguments from subcommand passthrough.
21pub fn run_vec(profile: &str, mut passthrough: Vec<String>) -> Box<run::Args> {
22    let mut command: Vec<String> = vec!["antimony", "run", profile]
23        .into_iter()
24        .map(String::from)
25        .collect();
26
27    command.append(&mut passthrough);
28    let cli = Cli::parse_from(command);
29    match cli.command {
30        Command::Run(args) => args,
31        _ => unreachable!(),
32    }
33}

58pub enum Command {

65    /// Edit an existing profile

71    /// Modify the system features.

74    /// Refresh caches

76
77    /// Integrate a profile into the user environment.
78    Integrate(integrate::Args),
79

82
83    /// Trace a profile for missing syscalls or files.
84    Trace(trace::Args),
85

90    Seccomp(seccomp::Args),
91
92    /// Export user profiles.
93    Export(export::Args),
94
95    /// Import user profiles.
96    Import(import::Args),
97
98    /// Import user profiles.
99    Config(config::Args),

101impl Run for Command {
102    fn run(self) -> Result<()> {
103        match self {
104            Command::Run(args) => args.run(),
105            Command::Create(args) => args.run(),
106            Command::Edit(args) => args.run(),
107            Command::Default(args) => args.run(),
108            Command::Feature(args) => args.run(),
109            Command::Refresh(args) => args.run(),
110            Command::Integrate(args) => args.run(),
111            Command::Reset(args) => args.run(),
112            Command::Trace(args) => args.run(),
113            Command::Info(args) => args.run(),
114            Command::Seccomp(args) => args.run(),
115            Command::Export(args) => args.run(),
116            Command::Import(args) => args.run(),
117            Command::Config(args) => args.run(),
118        }
119    }
120}
121impl Default for Command {
122    fn default() -> Self {
123        Self::Run(Box::default())
124    }
125}
