1//! Edit the default profile
2use crate::shared::{
3    env::AT_HOME,
4    profile::{self, Profile},
5};
6use anyhow::Result;
7use log::{error, trace};
8use std::fs;
9use user::as_effective;
10
11#[derive(clap::Args, Debug, Default)]
12pub struct Args {}
13
14impl super::Run for Args {
15    fn run(self) -> Result<()> {
16        let path = {
17            let path = Profile::default_profile();
18            if let Some(parent) = path.parent()
19                && !parent.exists()
20            {
21                fs::create_dir_all(parent)?;
22            }
23
24            if !path.exists() {
25                as_effective!({ fs::copy(AT_HOME.join("config").join("default.toml"), &path) })??;
26            }
27            path
28        };
29
30        trace!("Editing");
31        if let Err(e) = Profile::edit(&path) {
32            error!("Failed to edit default: {e}");
33            as_effective!(fs::remove_file(&path))??;
34            return Err(e.into());
35        } else {
36            as_effective!({ fs::remove_dir_all(profile::CACHE_DIR.as_path()) })??;
37        }
38        Ok(())
39    }
40}
