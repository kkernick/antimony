10pub fn setup(args: &mut super::Args) -> Result<Vec<String>> {
11    debug!("Setting up post arguments");
12    let mut post_args = Vec::new();
13
14    if let Some(mut arguments) = args.profile.lock().arguments.take() {
15        post_args.append(&mut arguments);
16    }
17    if let Some(passthrough) = &args.args.passthrough {
18        post_args.extend(passthrough.iter().cloned());
19    }
20
21    if !post_args.is_empty() {
22        let operation = match args.profile.lock().files.take() {
23            Some(mut files) => match files.passthrough.take() {
24                Some(passthrough) => passthrough,
25                None => FileMode::ReadOnly,
26            },
27            None => FileMode::ReadOnly,
28        };
29
30        for arg in &mut post_args {
31            let abs_arg = if as_real!(PWD.join(&arg).exists())? {
32                PWD.join(&arg).to_string_lossy().into_owned()
33            } else {
34                arg.to_string()
35            };
36
37            if as_real!(Path::new(&abs_arg).exists())? || abs_arg.starts_with("file://") {
38                let file = arg.strip_prefix("file://").unwrap_or(&abs_arg);
39                let dest = file.replace(HOME.as_str(), "/home/antimony");
40                match operation {
41                    FileMode::ReadOnly => args.handle.args_i(["--ro-bind", file, &dest])?,
42                    FileMode::ReadWrite => args.handle.args_i(["--bind", file, &dest])?,
43                    FileMode::Executable => {
44                        let contents = as_real!(fs::read_to_string(file))??;
45                        super::files::add_file(&args.handle, file, contents, FileMode::Executable)?
46                    }
47                };
48                *arg = dest;
49            }
50        }
51
52        return Ok(post_args);
53    }
54
55    Ok(Vec::new())
56}
