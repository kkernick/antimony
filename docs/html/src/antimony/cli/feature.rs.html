1//! Edit an existing profile.
2use crate::shared::{env::AT_HOME, feature::Feature, privileged, profile};
3use anyhow::{Result, anyhow};
4use dialoguer::Confirm;
5use std::fs;
6
7#[derive(clap::Args, Debug, Default)]
8pub struct Args {
9    /// The name of the feature
10    pub feature: String,
11
12    /// Delete a feature. It cannot be recovered. DANGEROUS!
13    #[arg(short, long, default_value_t = false)]
14    pub delete: bool,
15}
16
17impl super::Run for Args {
18    fn run(self) -> Result<()> {
19        if !privileged()? {
20            return Err(anyhow!(
21                "Modifying the system feature set is a privileged operation"
22            ));
23        } else {
24            user::set(user::Mode::Effective)?;
25
26            // Edit the feature
27            let feature = AT_HOME
28                .join("features")
29                .join(&self.feature)
30                .with_extension("toml");
31
32            let new = !feature.exists();
33            if new {
34                if self.delete {
35                    return Err(anyhow!("Requested feature does not exist!"));
36                }
37                if let Some(parent) = feature.parent() {
38                    fs::create_dir_all(parent)?;
39                }
40                fs::copy(AT_HOME.join("config").join("feature.toml"), &feature)?;
41            } else if self.delete {
42                let confirm = Confirm::new()
43                    .with_prompt(format!("Are you sure you want to delete {}?", self.feature))
44                    .interact()?;
45                if confirm {
46                    println!("Deleting {}", feature.display());
47                    fs::remove_file(&feature)?;
48                }
49                return Ok(());
50            }
51
52            // Edit it.
53            if Feature::edit(&feature)?.is_none() && new {
54                // If there was no modifications, delete the empty feature
55                fs::remove_file(feature)?;
56            } else {
57                fs::remove_dir_all(profile::CACHE_DIR.as_path())?;
58            }
59        }
60        Ok(())
61    }
62}
