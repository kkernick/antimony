1use anyhow::{Result, anyhow};
2use dashmap::DashSet;
3use inotify::{Inotify, WatchDescriptor};
4use log::debug;
5use spawn::Spawner;
6
7use crate::timer;
8
9pub fn setup(
10    watches: DashSet<WatchDescriptor>,
11    mut inotify: Inotify,
12    handle: &mut Spawner,
13    dry: bool,
14) -> Result<()> {
15    if let Some(proxy) = handle.get_associate("proxy")
16        && proxy.alive()?.is_none()
17    {
18        return Err(anyhow!("Proxy died!"));
19    }
20
21    timer!("::inotify", {
22        if !watches.is_empty() && !dry {
23            debug!("Waiting for inotify");
24            let mut buffer = [0; 1024];
25            while !watches.is_empty() {
26                let events = inotify.read_events_blocking(&mut buffer)?;
27                for event in events {
28                    if watches.contains(&event.wd) {
29                        if let Some(path) = event.name {
30                            debug!("Finished Notify Event: {}", path.display());
31                        }
32                        watches.remove(&event.wd);
33                    }
34                }
35            }
36        }
37    });
38    Ok(())
39}
