24fn format_iter<T, V>(iter: T) -> String
25where
26    T: Iterator<Item = V>,
27    V: std::fmt::Display,
28{
29    let mut ret = String::new();
30    iter.for_each(|f| ret.push_str(&format!("{f} ")));
31    ret
